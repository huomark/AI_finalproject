#include <bits/stdc++.h>
 using i64 = long long;
using u64 = unsigned long long;
using u32 = unsigned;
using u128 = unsigned __int128;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<i64> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
        int x = std::min_element(a.begin(), a.end()) - a.begin();
    i64 g = 0;
    for (int i = 0; i < n; i++) {
        if (i != x && a[i] % a[x] == 0) {
            g = std::gcd(g, a[i]);
        }
    }
    if (g == a[x]) {
        std::cout << "Yes\n";
    } else {
        std::cout << "No\n";
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
