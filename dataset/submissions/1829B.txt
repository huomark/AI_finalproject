#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        int ans = 0;
    int t = 0;
    for (int i = 0; i < n; i++) {
        int x;
        std::cin >> x;
        if (x == 0) {
            t++;
        } else {
            t = 0;
        }
        ans = std::max(ans, t);
    }
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
