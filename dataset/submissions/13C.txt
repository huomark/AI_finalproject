#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int N;
    std::cin >> N;
        std::priority_queue<int> pq;
        std::vector<int> a(N);
    for (int i = 0; i < N; i++) {
        std::cin >> a[i];
    }
        i64 ans = 0;
    for (int i = 0; i < N; i++) {
        if (pq.empty() || a[i] > pq.top()) {
            pq.push(a[i]);
        } else {
            ans += pq.top() - a[i];
            pq.push(a[i]);
            pq.push(a[i]);
            pq.pop();
        }
    }
        std::cout << ans << "\n";
        return 0;
}
