#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n, m;
    std::cin >> n >> m;
        std::vector<int> a(n * m);
    for (int i = 0; i < n * m; i++) {
        std::cin >> a[i];
    }
    std::sort(a.begin(), a.end(), std::greater());
        int ans = std::max({a[0] * (n * m - 1) - a[n * m - 1] * (n * m - n) - a[n * m - 2] * (n - 1),
        a[0] * (n * m - 1) - a[n * m - 1] * (n * m - m) - a[n * m - 2] * (m - 1),
        a[0] * (n * m - n) + a[1] * (n - 1) - a[n * m - 1] * (n * m - 1),
        a[0] * (n * m - m) + a[1] * (m - 1) - a[n * m - 1] * (n * m - 1)});
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
