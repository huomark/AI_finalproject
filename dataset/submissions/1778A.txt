#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> a(n);
    int sum = 0, min = 2;
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
        sum += a[i];
        if (i > 0) {
            min = std::min(min, a[i] + a[i - 1]);
        }
    }
    std::cout << sum - 2 * min << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
