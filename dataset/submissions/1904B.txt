#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
        std::vector<int> p(n);
    std::iota(p.begin(), p.end(), 0);
    std::sort(p.begin(), p.end(),
        [&](int i, int j) {
            return a[i] < a[j];
        });
        std::vector<int> ans(n);
    i64 s = 0;
    for (int i = 0, j = 0; i < n; i++) {
        if (j == i) {
            j++;
            s += a[p[i]];
        }
        while (j < n && s >= a[p[j]]) {
            s += a[p[j]];
            j++;
        }
        ans[p[i]] = j - 1;
    }
    for (int i = 0; i < n; i++) {
        std::cout << ans[i] << " \n"[i == n - 1];
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
