#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n, k;
    std::cin >> n >> k;
        std::vector<int> a(n + 2);
    std::priority_queue<i64, std::vector<i64>, std::greater<i64>> q;
    i64 add = 0;
    for (int i = 1; i <= n; i++) {
        std::cin >> a[i];
        if (a[i] == k) {
            a[i] = 0;
        }
    }
        i64 ans = 0;
    int p = 0;
    for (int i = 1; i <= n + 1; i++) {
        if (a[i] >= a[i - 1]) {
            q.push(a[i] - a[i - 1]);
        } else {
            q.push(0);
            q.push(k + a[i] - a[i - 1]);
        }
        ans += q.top();
        q.pop();
    }
        std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
