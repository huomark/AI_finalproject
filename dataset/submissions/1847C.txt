#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
        std::vector<int> s(n + 1);
    int ans = 0;
    for (int i = 0; i < n; i++) {
        s[i + 1] = s[i] ^ a[i];
    }
        std::sort(s.begin(), s.end());
        s.erase(std::unique(s.begin(), s.end()), s.end());
        for (int i = 0; i < s.size(); i++) {
        for (int j = 0; j < i; j++) {
            ans = std::max(ans, s[i] ^ s[j]);
        }
    }
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
    while (t--) {
        solve();
    }
        return 0;
}
