#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n;
    std::cin >> n;
        std::vector<int> f(n), p(n);
    for (int i = 0; i < n; i++) {
        std::cin >> f[i];
        f[i]--;
        p[f[i]] = i;
    }
        i64 ans = 0;
    for (int i = 1; i < n; i++) {
        ans += std::abs(p[i-1] - p[i]);
    }
    std::cout << ans << "\n";
        return 0;
}
