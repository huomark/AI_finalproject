#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n;
    std::cin >> n;
        std::vector a(n, std::vector<int>(n));
        for (int x = 0; x < n - 1; x++) {
        for (int j = 0; j < n - 1; j++) {
            int k = (x - j + n - 1) % (n - 1);
            if (j == k) {
                a[j][n - 1] = a[n - 1][j] = x + 1;
            } else {
                a[j][k] = a[k][j] = x + 1;
            }
        }
    }
        for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            std::cout << a[i][j] << " \n"[j == n - 1];
        }
    }
        return 0;
}
