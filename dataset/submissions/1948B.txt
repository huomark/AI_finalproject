#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
        int lst = 99;
    int ok = 1;
    for (int i = n - 1; i >= 0; i--) {
        if (a[i] > lst) {
            if (a[i] < 10) {
                ok = 0;
            } else {
                int x = a[i] / 10;
                int y = a[i] % 10;
                if (x > y || y > lst) {
                    ok = 0;
                }
                lst = x;
            }
        } else {
            lst = a[i];
        }
    }
    std::cout << (ok ? "YES" : "NO") << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
