#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n;
    std::cin >> n;
        std::string s;
    std::cin >> s;
        std::vector<int> pre(2 * n);
    int ans = 0;
    for (int i = 0; i < 2 * n - 1; i++) {
        pre[i + 1] = pre[i] + (s[i] == 'W');
    }
    for (int i = 0; i < n; i++) {
        ans = std::max(ans, pre[i + n] - pre[i]);
    }
        std::cout << ans << "\n";
        return 0;
}
