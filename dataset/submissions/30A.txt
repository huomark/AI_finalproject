#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int A, B, n;
    std::cin >> A >> B >> n;
        if (B == 0) {
        std::cout << 0 << "\n";
        return 0;
    }
        if (A == 0) {
        std::cout << "No solution\n";
        return 0;
    }
        int v = B / A, x;
    if (v < 0) {
        v = -v;
        if (n % 2 == 0) {
            std::cout << "No solution\n";
            return 0;
        }
        x = std::pow(v, 1. / n) + .5;
        x = -x;
    } else {
        x = std::pow(v, 1. / n) + .5;
    }
        for (int i = 0; i < n; i++) {
        A *= x;
    }
    if (A != B) {
        std::cout << "No solution\n";
    } else {
        std::cout << x << "\n";
    }
        return 0;
}
