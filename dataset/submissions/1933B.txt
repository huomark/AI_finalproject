#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> a(n);
    int sum = 0;
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
        sum += a[i];
    }
        if (sum % 3 == 0) {
        std::cout << 0 << "\n";
        return;
    }
        if (sum % 3 == 2) {
        std::cout << 1 << "\n";
        return;
    }
        for (int i = 0; i < n; i++) {
        if (sum % 3 == a[i] % 3) {
            std::cout << 1 << "\n";
            return;
        }
    }
    std::cout << 2 << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
