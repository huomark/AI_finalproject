#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n, m;
    std::cin >> n >> m;
        std::vector<int> l(n), r(n), len(n);
    for (int i = 0; i < n; i++) {
        std::cin >> l[i] >> r[i];
        l[i]--;
        len[i] = r[i] - l[i];
    }
        int ans = 0;
        int max = *std::max_element(len.begin(), len.end());
    int min = *std::min_element(len.begin(), len.end());
    ans = std::max(ans, max - min);
        int maxl = *std::max_element(l.begin(), l.end());
    int minr = *std::min_element(r.begin(), r.end());
        for (int i = 0; i < n; i++) {
        ans = std::max(ans, std::min(len[i], std::max(maxl - l[i], r[i] - minr)));
    }
        std::cout << 2 * ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
