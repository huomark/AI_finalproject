#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        int cnt = 0, ans = 0;
    int avi = 0;
    for (int i = 0; i < n; i++) {
        int b;
        std::cin >> b;
                if (b == 1) {
            cnt += 1;
            avi += 1;
            ans = std::max(ans, avi);
        } else {
            if (cnt > 0) {
                avi = cnt / 2 + 1;
            }
        }
    }
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
