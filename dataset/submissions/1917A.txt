#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> a(n);
    int sgn = 1;
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
        if (a[i] < 0) {
            sgn *= -1;
        } else if (a[i] == 0) {
            sgn = 0;
        }
    }
        if (sgn <= 0) {
        std::cout << 0 << "\n";
        return;
    }
    std::cout << 1 << "\n";
    std::cout << 1 << " " << 0 << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
