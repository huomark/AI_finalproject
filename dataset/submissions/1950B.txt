#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        for (int i = 0; i < 2 * n; i++) {
        for (int j = 0; j < 2 * n; j++) {
            std::cout << "#."[(i >> 1 ^ j >> 1) & 1];
        }
        std::cout << "\n";
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
