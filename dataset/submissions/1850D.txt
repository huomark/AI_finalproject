#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n, k;
    std::cin >> n >> k;
        std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
    int ans = 0;
        std::sort(a.begin(), a.end());
        int lst = 0;
    for (int i = 1; i <= n; i++) {
        if (i == n || a[i] - a[i - 1] > k) {
            ans = std::max(ans, i - lst);
            lst = i;
        }
    }
    std::cout << n - ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
