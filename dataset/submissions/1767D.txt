#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n;
    std::cin >> n;
        std::string s;
    std::cin >> s;
        int lt = 0, gt = 0;
        for (int i = 0; i < n; i++) {
        if (s[i] == '1') lt += lt + 1;
        else gt += gt + 1;
    }
        for (int i = lt + 1; i <= (1 << n) - gt; i++) {
        std::cout << i << " \n"[i == n - gt];
    }
        return 0;
}
