#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> cnt(n + 1);
    std::vector<std::pair<int, int>> c(n);
    for (int i = 0; i < n; i++) {
        std::cin >> c[i].first;
    }
    for (int i = 0; i < n; i++) {
        std::cin >> c[i].second;
    }
    i64 ans = 0;
    std::sort(c.begin(), c.end());
    for (int s = 1; s * s <= 2 * n; s++) {
        cnt.assign(n + 1, 0);
        for (auto [a, b] : c) {
            int v = a * s - b;
            if (1 <= v && v <= n) {
                ans += cnt[v];
            }
            if (a == s) {
                cnt[b]++;
            }
        }
    }
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
