#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        std::string s;
    std::cin >> s;
        const int n = s.size();
        std::vector<int> p[26];
    for (int i = 0; i < 26; i++) {
        p[i].push_back(-1);
    }
    for (int i = 0; i < n; i++) {
        p[s[i] - 'a'].push_back(i);
    }
    for (int i = 0; i < 26; i++) {
        p[i].push_back(n);
    }
        int ans = n;
    for (int i = 0; i < 26; i++) {
        int res = 0;
        for (int j = 1; j < p[i].size(); j++) {
            res = std::max(res, p[i][j] - p[i][j - 1]);
        }
        ans = std::min(ans,res);
    }
    std::cout << ans << "\n";
        return 0;
}
