#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n, m;
    std::cin >> n >> m;
        std::vector<int> a(m);
    i64 ans = 1LL * n * (n-1) / 2;
    for (int i = 0; i < n; i++) {
        int x;
        std::cin >> x;
        x--;
        ans -= a[x]++;
    }
    std::cout << ans << "\n";
        return 0;
}
