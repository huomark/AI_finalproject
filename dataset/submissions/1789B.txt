#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::string s;
    std::cin >> s;
        int l = 0, r = n;
    while (l < r && s[l] == s[r - 1]) {
        l++, r--;
    }
    if (l >= r) {
        std::cout << "Yes\n";
        return;
    }
    while (l < r && s[l] != s[r - 1]) {
        l++, r--;
    }
    while (l < r && s[l] == s[r - 1]) {
        l++, r--;
    }
    if (l >= r) {
        std::cout << "Yes\n";
    } else {
        std::cout << "No\n";
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
