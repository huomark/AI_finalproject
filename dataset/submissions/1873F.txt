#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n, k;
    std::cin >> n >> k;
        std::vector<int> s(n + 1), h(n);
    for (int i = 0; i < n; i++) {
        int a;
        std::cin >> a;
        s[i + 1] = s[i] + a;
    }
    for (int i = 0; i < n; i++) {
        std::cin >> h[i];
    }
        int ans = 0;
    for (int i = 0, j = 0; i < n; i++) {
        if (i && h[i - 1] % h[i] != 0) {
            j = i;
        }
        while (s[i + 1] - s[j] > k) {
            j++;
        }
        ans = std::max(ans, i - j + 1);
    }
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
