#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        int ans = 0;
        int t = 0;
    for (int i = 0; i < n; i++) {
        int x;
        std::cin >> x;
        ans += (x - t) / 120;
        t = x;
    }
    ans += (1440 - t) / 120;
    std::cout << (ans >= 2 ? "YES" : "NO") << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
