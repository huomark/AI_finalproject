#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n, m;
    std::cin >> n >> m;
        std::vector<int> a(m + 1);
    a[0] = 1;
    for (int i = 1; i <= m; i++) {
        std::cin >> a[i];
    }
        i64 ans = 0;
    for (int i = 1; i <= m; i++) {
        int d = (a[i] - a[i - 1] + n) % n;
        ans += d;
    }
    std::cout << ans << "\n";
        return 0;
}
