 using i64 = long long;

 void solve() {
    int n, m, k;
    std::cin >> n >> m >> k;
        if (n <= (m + 1) * k) {
        for (int i = 0; i < n; i++) {
            std::cout << i % k << " \n"[i == n - 1];
        }
    } else {
        int ans = n / (m + 1);
        for (int i = 0; i < n; i++) {
            std::cout << i % ans << " \n"[i == n - 1];
        }
    }
}
