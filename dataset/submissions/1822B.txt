#include <bits/stdc++.h> 
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
    std::sort(a.begin(), a.end());
        if (n > 4) {
        a.erase(a.begin() + 2, a.end() - 2);
        n = 4;
    }
        i64 ans = -1E18;
    for (int i = 0; i < n; i++) {
        for (int j = i+1; j < n; j++) {
            ans = std::max(ans, 1LL * a[i] * a[j]);
        }
    }
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
