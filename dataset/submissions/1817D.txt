 using i64 = long long;
 using namespace std::literals::string_literals;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n, k;
    std::cin >> n >> k;
        n -= 2;
        std::string s;
    if (k == (n + 1) / 2) {
        s = "DL"s;

        for (int i = 0; i < k - 1; i++) {
            s += "LDRU";
        }

        for (int i = 0; i < n - k; i++) {

            s += "LDLU";
        }
        s += "RDL";
    }
    std::cout << s << "\n";

}
