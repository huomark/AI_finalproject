#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n, s1, s2;
    std::cin >> n >> s1 >> s2;
        std::vector<int> r(n);
    for (int i = 0; i < n; i++) {
        std::cin >> r[i];
    }
        std::vector<int> p(n);
    std::iota(p.begin(), p.end(), 0);
    std::sort(p.begin(), p.end(), [&](int i, int j) {
        return r[i] > r[j];
    });
        std::vector<int> a, b;
    for (auto i : p) {
        if ((a.size() + 1) * s1 < (b.size() + 1) * s2) {
            a.push_back(i + 1);
        } else {
            b.push_back(i + 1);
        }
    }
        std::cout << a.size();
    for (auto x : a) {
        std::cout << " " << x;
    }
    std::cout << "\n";
    std::cout << b.size();
    for (auto x : b) {
        std::cout << " " << x;
    }
    std::cout << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
