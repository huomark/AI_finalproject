#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int N, M;
    std::cin >> N >> M;
        constexpr int B = 400;
        std::vector<int> a(N), b(N), c(N), d(N);
    for (int i = 0; i < N; i++) {
        std::cin >> a[i];
        a[i] += i;
    }
    for (int i = N - 1; i >= 0; i--) {
        if (a[i] >= N || a[i] / B != i / B) {
            b[i] = i;
            c[i] = 1;
        } else {
            b[i] = b[a[i]];
            c[i] = 1 + c[a[i]];
        }
    }
        while (M--) {
        int t;
        std::cin >> t;
                if (t == 0) {
            int x, y;
            std::cin >> x >> y;
            x--;
            a[x] = x + y;
            for (int i = x; i >= x / B * B; i--) {
                if (a[i] >= N || a[i] / B != i / B) {
                    b[i] = i;
                    c[i] = 1;
                } else {
                    b[i] = b[a[i]];
                    c[i] = 1 + c[a[i]];
                }
            }
        } else {
            int x;
            std::cin >> x;
            x--;
            int y = -1, cnt = 0;
            while (x < N) {
                cnt += c[x];
                y = b[x];
                x = a[b[x]];
            }
            std::cout << y + 1 << " " << cnt << "\n";
        }
    }
        return 0;
}
