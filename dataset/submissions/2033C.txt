#include <bits/stdc++.h>
 using i64 = long long;
using u64 = unsigned long long;
using u32 = unsigned;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
        int ans = 0;
    for (int i = 1; i <= n - 1 - i; i++) {
        int x = a[i];
        int y = a[n - 1 - i];
        int z = a[i - 1];
        int w = a[n - i];
        ans += std::min((x == z) + (y == w), (x == w) + (y == z));
    }
    if (n % 2 == 0) {
        ans += (a[n / 2 - 1] == a[n / 2]);
    }
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
