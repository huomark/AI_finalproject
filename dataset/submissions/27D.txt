#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n, m;
    std::cin >> n >> m;
        std::vector<int> a(m), b(m);
    for (int i = 0; i < m; i++) {
        std::cin >> a[i] >> b[i];
        a[i]--, b[i]--;
        if (a[i] > b[i]) {
            std::swap(a[i], b[i]);
        }
    }
        std::vector<std::vector<int>> adj(m);
    for (int i = 0; i < m; i++) {
        for (int j = 0; j < m; j++) {
            if (a[i] < a[j] && a[j] < b[i] && b[i] < b[j]) {
                adj[i].push_back(j);
                adj[j].push_back(i);
            }
        }
    }
        std::vector<int> c(m, -1);
    for (int i = 0; i < m; i++) {
        if (c[i] == -1) {
            std::queue<int> q;
            q.push(i);
            c[i] = 0;
                        while (!q.empty()) {
                int x = q.front();
                q.pop();
                                for (auto y : adj[x]) {
                    if (c[y] == -1) {
                        c[y] = !c[x];
                        q.push(y);
                    } else if (c[x] == c[y]) {
                        std::cout << "Impossible\n";
                        return 0;
                    }
                }
            }
        }
    }
        for (int i = 0; i < m; i++) {
        std::cout << "io"[c[i]];
    }
    std::cout << "\n";
        return 0;
}
