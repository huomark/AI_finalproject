#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> a(n), b(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
    for (int i = 0; i < n; i++) {
        std::cin >> b[i];
    }
        int t = -1;
    for (int i = 0; i < n; i++) {
        if (a[i] == 0 && b[i] == 0) {
            continue;
        }
                int v = 0;
        int x = a[i], y = b[i];
        while (x != 0) {
            y %= (2 * x);
            std::tie(x, y) = std::make_pair(y, std::abs(x - y));
            v = (v + 1) % 3;
        }
                if (t != -1 && t != v) {
            std::cout << "NO\n";
            return;
        }
        t = v;
    }
        std::cout << "YES\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
