#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n, m;
    std::cin >> n >> m;
        std::vector<std::pair<int, int>> a(m);
    for (int i = 0; i < m; i++) {
        std::cin >> a[i].second >> a[i].first;
    }
    std::sort(a.begin(), a.end(), std::greater());
    int ans = 0;
    for (auto [v, c] : a) {
        int t = std::min(c, n);
        n -= t;
        ans += v * t;
    }
    std::cout << ans << "\n";
        return 0;
}
