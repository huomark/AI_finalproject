#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int k, a, b;
    std::cin >> k >> a >> b;
    std::string s;
    std::cin >> s;
        int n = s.size();
    if (n < a * k || n > b * k) {
        std::cout << "No solution\n";
        return 0;
    }
        for (int i = 0, j = 0; i < k; i++) {
        int t = std::min(b, n - j - (k-i-1) * a);
        std::cout << s.substr(j, t) << "\n";
        j += t;
    }
        return 0;
}
