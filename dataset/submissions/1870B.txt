#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n, m;
    std::cin >> n >> m;
        std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
        int o = 0;
    for (int i = 0; i < m; i++) {
        int b;
        std::cin >> b;
        o |= b;
    }
        int x = 0, y = 0;
    for (int i = 0; i < n; i++) {
        x ^= a[i];
        y ^= (a[i] | o);
    }
    if (x > y) {
        std::swap(x, y);
    }
    std::cout << x << " " << y << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
