#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> a(n), b(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
    for (int i = 0; i < n; i++) {
        std::cin >> b[i];
    }
        int l = 0, r = n - 1;
    while (a[l] == b[l]) {
        l++;
    }
    while (a[r] == b[r]) {
        r--;
    }
        while (l && b[l] >= b[l - 1]) {
        l--;
    }
    while (r + 1 < n && b[r] <= b[r + 1]) {
        r++;
    }
    std::cout << l + 1 << " " << r + 1 << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
