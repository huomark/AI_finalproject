#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n, q;
    std::cin >> n >> q;
        std::vector<std::pair<int, int>> a(n);
    for (int i = 0; i < n; i++) {
        int x, y;
        std::cin >> x >> y;
                a[i] = {x, y};
    }
        std::sort(a.begin(), a.end());
        std::vector<std::pair<int, int>> b;
    for (auto [x, y] : a) {
        if (!b.empty() && b.back().first == x) {
            continue;
        }
        while (!b.empty() && b.back().second >= y) {
            b.pop_back();
        }
        b.emplace_back(x, y);
    }
    a = std::move(b);
        n = a.size();
        int lg = std::__lg(n);
        std::vector f(lg + 1, std::vector<int>(n, n));
    for (int i = 0, j = 0; i < n; i++) {
        while (j < n && a[j].first <= a[i].second) {
            j++;
        }
        f[0][i] = j;
    }
    for (int k = 0; k < lg; k++) {
        for (int i = 0; i < n; i++) {
            if (f[k][i] < n) {
                f[k + 1][i] = f[k][f[k][i]];
            }
        }
    }
        for (int i = 0; i < q; i++) {
        int l, r;
        std::cin >> l >> r;
                auto x = std::lower_bound(a.begin(), a.end(), std::pair(l, -1)) - a.begin();
        if (x == n || a[x].first != l) {
            std::cout << 0 << "\n";
            continue;
        }
                if (a[x].second > r) {
            std::cout << 0 << "\n";
            continue;
        }
                int y = x + 1;
                if (y == n || a[y].second > r) {
            if (a[x].second == r) {
                std::cout << 998244352 << "\n";
            } else {
                std::cout << 0 << "\n";
            }
            continue;
        }
                for (int j = lg; j >= 0; j--) {
            if (f[j][x] < n && a[f[j][x]].second <= r) {
                x = f[j][x];
            }
            if (f[j][y] < n && a[f[j][y]].second <= r) {
                y = f[j][y];
            }
        }
                if (x == y) {
            std::cout << 0 << "\n";
        } else if (a[x].second == r) {
            std::cout << 998244352 << "\n";
        } else if (a[y].second == r) {
            std::cout << 1 << "\n";
        } else {
            std::cout << 0 << "\n";
        }
    }
        return 0;
}
