#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n, k;
    std::cin >> n >> k;
        std::vector<int> c(n);
    for (int i = 0; i < n; i++) {
        std::cin >> c[i];
    }
        if (c[0] == c[n - 1]) {
        if (std::count(c.begin(), c.end(), c[0]) >= k) {
            std::cout << "YES\n";
        } else {
            std::cout << "NO\n";
        }
        return;
    }
        int c1 = 0;
    int c2 = std::count(c.begin(), c.end(), c[n - 1]);
    for (int i = 0; i < n; i++) {
        c1 += (c[0] == c[i]);
        c2 -= (c[n - 1] == c[i]);
        if (c1 >= k && c2 >= k) {
            std::cout << "YES\n";
            return;
        }
    }
    std::cout << "NO\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
