#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        int pos = 0, neg = 0;
    for (int i = 0; i < n; i++) {
        int b;
        std::cin >> b;
        if (b > 0) {
            pos += 1;
        } else {
            neg += 1;
        }
    }
        for (int i = 1; i <= n; i++) {
        int p = std::min(i, pos);
        std::cout << p - (i - p) << " \n"[i == n];
    }
    for (int i = 1; i <= n; i++) {
        std::cout << std::max(i % 2, pos - neg - (n - i)) << " \n"[i == n];
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
