#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n, t;
    std::cin >> n >> t;
        std::string s;
    std::cin >> s;
        for (int i = 0; i < t; i++) {
        for (int j = 1; j < n; j++) {
            if (s[j - 1] == 'B' && s[j] == 'G') {
                std::swap(s[j - 1], s[j]);
                j++;
            }
        }
    }
        std::cout << s << "\n";
        return 0;
}
