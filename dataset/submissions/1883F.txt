#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
        std::map<int, int> first, last;
    for (int i = 0; i < n; i++) {
        if (!first.count(a[i])) {
            first[a[i]] = i;
        }
        last[a[i]] = i;
    }
    std::vector<int> l(n), r(n);
    for (auto [x, y] : first) {
        l[y] = 1;
    }
    for (auto [x, y] : last) {
        r[y] = 1;
    }
        i64 ans = 0;
    int cnt = 0;
    for (int i = 0; i < n; i++) {
        cnt += l[i];
        ans += cnt * r[i];
    }
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
