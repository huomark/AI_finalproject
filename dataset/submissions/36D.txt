#include <bits/stdc++.h>
 using i64 = long long;
    #ifdef ONLINE_JUDGE
    std::ifstream fin("input.txt");
    std::ofstream fout("output.txt");
#else
    #define fin std::cin
    #define fout std::cout
#endif
 int main() {
        int t, k;
    fin >> t >> k;
        while (t--) {
        int n, m;
        fin >> n >> m;
        n--, m--;
                int ans;
        if (std::min(n, m) % (k + 1) == k) {
            ans = 1;
        } else {
            ans = (n + m - std::min(n, m) / (k + 1) * (k != 1)) % 2;
        }
        fout << "-+"[ans] << "\n";
    }
        return 0;
}
