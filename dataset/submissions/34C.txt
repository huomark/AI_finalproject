#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        std::string s;
    std::cin >> s;
        std::replace(s.begin(), s.end(), ',', ' ');
        std::stringstream str;
    str << s;
        int x;
    std::vector<int> a;
    while (str >> x) {
        a.push_back(x);
    }
        std::sort(a.begin(), a.end());
    a.erase(std::unique(a.begin(), a.end()), a.end());
        int n = a.size();
        bool first = true;
    for (int i = 0; i < n; i++) {
        int j = i;
        while (j + 1 < n && a[j + 1] == a[j] + 1) {
            j++;
        }
        if (first) {
            first = false;
        } else {
            std::cout << ',';
        }
        std::cout << a[i];
        if (i < j) {
            std::cout << '-' << a[j];
        }
        i = j;
    }
    std::cout << "\n";
        return 0;
}
