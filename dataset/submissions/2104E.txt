#include <bits/stdc++.h>
 using i64 = long long;
using u64 = unsigned long long;
using u32 = unsigned;
using u128 = unsigned __int128;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n, k;
    std::cin >> n >> k;
        std::string s;
    std::cin >> s;
        std::vector nxt(k, std::vector<int>(n + 2, n + 1));
    for (int x = 0; x < k; x++) {
        for (int i = n - 1; i >= 0; i--) {
            nxt[x][i] = s[i] == 'a' + x ? i + 1 : nxt[x][i + 1];
        }
    }
        std::vector<int> f(n + 2);
    for (int i = n; i >= 0; i--) {
        f[i] = 1E9;
        for (int x = 0; x < k; x++) {
            f[i] = std::min(f[i], 1 + f[nxt[x][i]]);
        }
    }
        int q;
    std::cin >> q;
        for (int i = 0; i < q; i++) {
        std::string t;
        std::cin >> t;
                int x = 0;
        for (auto c : t) {
            x = nxt[c - 'a'][x];
        }
        std::cout << f[x] << "\n";
    }
        return 0;
}
