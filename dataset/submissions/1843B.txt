#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        i64 sum = 0;
    int ans = 0;
    int last = 1;
    std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
        sum += std::abs(a[i]);
        if (a[i] < 0 && last == 1) {
            ans++;
            last = -1;
        } else if (a[i] > 0) {
            last = 1;
        }
    }
        std::cout << sum << " " << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
