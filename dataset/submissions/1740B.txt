#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> a(n), b(n);
    i64 ans = 0;
    for (int i = 0; i < n; i++) {
        std::cin >> a[i] >> b[i];
        if (a[i] > b[i]) {
            std::swap(a[i], b[i]);
        }
        ans += a[i];
    }
    ans += *std::max_element(b.begin(), b.end());
    ans *= 2;
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
