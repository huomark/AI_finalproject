#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n, m;
    std::cin >> n >> m;
        std::vector<int> l(m), r(m);
    std::vector<std::vector<int>> f(n);
    for (int i = 0; i < m; i++) {
        std::cin >> l[i] >> r[i];
        l[i] = n - l[i];
        f[l[i]].push_back(i);
    }
        std::priority_queue<std::pair<int, int>, std::vector<std::pair<int, int>>, std::greater<>> q;
    std::vector<int> ans;
    for (int i = 0; i < n; i++) {
        for (auto j : f[i]) {
            q.emplace(r[j], j);
        }
        while (!q.empty()) {
            auto [r, j] = q.top();
            q.pop();
                        if (r <= i) {
                continue;
            }
                        ans.push_back(j);
            break;
        }
    }
        std::cout << ans.size() << "\n";
    for (int i = 0; i < ans.size(); i++) {
        std::cout << ans[i] + 1 << " \n"[i == ans.size() - 1];
    }
        return 0;
}
