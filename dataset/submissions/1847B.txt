#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        int all = ~0;
    std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
        all &= a[i];
    }
        if (all != 0) {
        std::cout << 1 << "\n";
        return;
    }
        int ans = 0;
    int res = ~0;
    for (int i = 0; i < n; i++) {
        res &= a[i];
        if (res == 0) {
            ans++;
            res = ~0;
        }
    }
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
    while (t--) {
        solve();
    }
        return 0;
}
