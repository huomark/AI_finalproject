#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n;
    std::cin >> n;
        std::vector a(n, std::vector<int>(n));
    i64 sum = 0;
    int min = 1E9;
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            std::cin >> a[i][j];
            sum += a[i][j];
            if (i + j == n - 1) {
                min = std::min(min, a[i][j]);
            }
        }
    }
    sum -= min;
    std::cout << sum << "\n";
        return 0;
}
