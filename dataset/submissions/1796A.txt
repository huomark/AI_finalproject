#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int k;
    std::cin >> k;
        std::string s;
    std::cin >> s;
        std::string t;
    for (int i = 1; i <= 100; i++) {
        if (i % 3 == 0) {
            t += 'F';
        }
        if (i % 5 == 0) {
            t += 'B';
        }
    }
    for (int i = 0; i + k <= t.size(); i++) {
        if (t.substr(i, k) == s) {
            std::cout << "YES\n";
            return;
        }
    }
        std::cout << "NO\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
