#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        if (n % 2 == 0) {
        std::cout << -1 << "\n";
        return;
    }
        std::vector<int> ans;
    while (n != 1) {
        if ((n + 1) / 2 % 2 == 1) {
            n += 1;
            ans.push_back(1);
        } else {
            n -= 1;
            ans.push_back(2);
        }
        n /= 2;
    }
        std::reverse(ans.begin(), ans.end());
    std::cout << ans.size() << "\n";
    for (int i = 0; i < ans.size(); i++) {
        std::cout << ans[i] << " \n"[i == ans.size() - 1];
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
