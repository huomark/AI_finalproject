#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    i64 n, x;
    std::cin >> n >> x;
        std::vector<i64> f(60);
    for (int i = 0; i < 60; i++) {
        if (~n >> i & 1) {
            f[i] = n;
        } else {
            f[i] = (n & ~((1LL << i) - 1)) + (1LL << i);
        }
    }
        i64 m = n;
    for (int i = 0; i < 60; i++) {
        if (~x >> i & 1) {
            m = std::max(m, f[i]);
        }
    }
    for (int i = 0; i < 60; i++) {
        if (x >> i & 1) {
            if (m >= f[i]) {
                std::cout << -1 << "\n";
                return;
            }
        }
    }
        std::cout << m << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
