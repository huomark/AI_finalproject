#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        int ans = 1E9;
    std::vector<std::pair<int, int>> a(2 * n);
    for (int i = 0; i < n; i++) {
        int x, y;
        std::cin >> x >> y;
                a[2 * i] = {x, 2 * i};
        a[2 * i + 1] = {y, 2 * i + 1};
    }
    std::sort(a.begin(), a.end());
        std::pair<int, int> mx[2][2];
    for (int i = 0; i < 2; i++) {
        for (int j = 0; j < 2; j++) {
            mx[i][j] = {-1, -1};
        }
    }
    std::vector<bool> vis(n);
    int bad = n;
    for (auto [v, t] : a) {
        int u = t % 2;
        mx[u][1] = mx[u][0];
        mx[u][0] = {v, t};
        bad -= !vis[t / 2];
        vis[t / 2] = true;
        for (int i = 0; i < 2; i++) {
            if (bad == 0 && mx[!u][i].second != -1 && mx[!u][i].second != (t ^ 1)) {
                ans = std::min(ans, v - mx[!u][i].first);
            }
        }
    }
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
