#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n;
    std::cin >> n;
        std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
        std::array<int, 3> dp {};
        for (int i = 0; i < n; i++) {
        for (int j = 0; j < 3; j++) {
            dp[j] += a[i] * (j == 1 ? 1 : -1);
        }
        for (int j = 1; j < 3; j++) {
            dp[j] = std::max(dp[j], dp[j - 1]);
        }
    }
    std::cout << dp[2] << "\n";
        return 0;
}
