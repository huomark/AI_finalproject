#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        int cnt[30] {};
    for (int i = 0; i < n; i++) {
        int a;
        std::cin >> a;
        for (int j = 0; j < 30; j++) {
            cnt[j] += a >> j & 1;
        }
    }
        int ans = 0;
    for (int i = 0; i < 30; i++) {
        ans = std::gcd(ans, cnt[i]);
    }
    for (int i = 1; i <= n; i++) {
        if (ans % i == 0) {
            std::cout << i << " ";
        }
    }
    std::cout << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
