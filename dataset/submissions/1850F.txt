#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> cnt(n + 1);
    for (int i = 0; i < n; i++) {
        int a;
        std::cin >> a;
                if (a <= n) {
            cnt[a]++;
        }
    }
        for (int i = n; i >= 1; i--) {
        for (int j = 2 * i; j <= n; j += i) {
            cnt[j] += cnt[i];
        }
    }
        auto ans = *std::max_element(cnt.begin(), cnt.end());
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
