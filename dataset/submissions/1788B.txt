#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        auto s = std::to_string(n);
        std::string a, b;
        int sum = 0;
    for (int i = 0; i < s.size(); i++) {
        int v = s[i] - '0';
        int x, y;
        if (sum > 0) {
            x = v / 2;
        } else {
            x = (v + 1) / 2;
        }
        y = v - x;
        a += '0' + x;
        b += '0' + y;
        sum += x - y;
    }
        int x = std::stoi(a);
    int y = std::stoi(b);
        std::cout << x << " " << y << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
