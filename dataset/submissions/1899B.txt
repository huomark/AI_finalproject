#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> a(n);
    std::vector<i64> s(n + 1);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
    for (int i = 0; i < n; i++) {
        s[i + 1] = s[i] + a[i];
    }
        i64 ans = 0;
    for (int k = 1; k <= n; k++) {
        if (n % k == 0) {
            i64 mx = 0, mn = s[n];
            for (int i = 0; i < n; i += k) {
                i64 v = s[i + k] - s[i];
                mx = std::max(mx, v);
                mn = std::min(mn, v);
            }
            ans = std::max(ans, mx - mn);
        }
    }
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
