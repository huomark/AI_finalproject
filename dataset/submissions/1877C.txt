#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n, m, k;
    std::cin >> n >> m >> k;
        if (k == 1) {
        std::cout << 1 << "\n";
    } else if (k == 2) {
        std::cout << std::min(m, n - 1) + m / n << "\n";
    } else if (k == 3) {
        std::cout << m - std::min(m, n - 1) - m / n << "\n";
    } else {
        std::cout << 0 << "\n";
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
