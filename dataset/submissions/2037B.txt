#include <bits/stdc++.h>
 using i64 = long long;
using u64 = unsigned long long;
using u32 = unsigned;
using u128 = unsigned __int128;
 void solve() {
    int k;
    std::cin >> k;
        std::vector<int> a(k);
    for (int i = 0; i < k; i++) {
        std::cin >> a[i];
    }
    std::vector<int> cnt(k + 1);
    for (auto x : a) {
        cnt[x]++;
    }
        for (int i = 1; i <= k - 2; i++) {
        if ((k - 2) % i == 0) {
            int j = (k - 2) / i;
            if (i == j ? cnt[i] >= 2 : (cnt[i] > 0 && cnt[j] > 0)) {
                std::cout << i << " " << j << "\n";
                return;
            }
        }
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
