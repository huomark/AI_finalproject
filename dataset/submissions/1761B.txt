#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
        int flag = n % 2 == 0;
    for (int i = 2; i < n; i++) {
        if (a[i] != a[i - 2]) {
            flag = 0;
        }
    }
        if (flag) {
        std::cout << n / 2 + 1 << "\n";
    } else {
        std::cout << n << "\n";
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
