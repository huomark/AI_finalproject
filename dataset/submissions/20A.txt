#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        std::string s;
    std::cin >> s;
        std::string t;
    for (auto c : s) {
        if (!t.empty() && t.back() == '/' && c == '/') {
            continue;
        }
        t += c;
    }
    while (t.size() > 1 && t.back() == '/') {
        t.pop_back();
    }
    std::cout << t << "\n";
        return 0;
}
