#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n;
    std::cin >> n;
        int res = 0;
    for (int i = 0; i < n; i++) {
        int m, c;
        std::cin >> m >> c;
        if (m > c) {
            res += 1;
        }
        if (m < c) {
            res -= 1;
        }
    }
        if (res > 0) {
        std::cout << "Mishka\n";
    } else if (res < 0) {
        std::cout << "Chris\n";
    } else {
        std::cout << "Friendship is magic!^^\n";
    }
        return 0;
}
