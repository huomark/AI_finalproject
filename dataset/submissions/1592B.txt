#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n, x;
    std::cin >> n >> x;
        std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
    if (n - x <= x - 1) {
        int l = n - x, r = x;
        std::rotate(a.begin() + l, a.begin() + r, a.end());
        std::sort(a.begin(), a.end() - (r - l));
        std::rotate(a.begin() + l, a.end() - (r - l), a.end());
        if (std::is_sorted(a.begin(), a.end())) {
            std::cout << "YES\n";
        } else {
            std::cout << "NO\n";
        }
    } else {
        std::cout << "YES\n";
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
