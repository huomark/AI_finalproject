#include <bits/stdc++.h>
 using i64 = long long;
    #ifdef ONLINE_JUDGE
    std::ifstream fin("input.txt");
    std::ofstream fout("output.txt");
#else
    #define fin std::cin
    #define fout std::cout
#endif
 int main() {
        int n;
    fin >> n;
        double ans = 0;
    std::vector<double> h(n), r(n), R(n);
    for (int i = 0; i < n; i++) {
        fin >> h[i] >> r[i] >> R[i];
    }
        std::vector<double> dp(n);
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < i; j++) {
            double x = 0;
            if (r[i] >= R[j]) {
                x = h[j];
            } else {
                x = std::max({0., (r[i] - r[j]) / (R[j] - r[j]) * h[j]
                    , std::min((R[i] - r[j]) / (R[j] - r[j]) * h[j] - h[i], h[j] - (R[j] - r[i]) / (R[i] - r[i]) * h[i])});
            }
            dp[i] = std::max(dp[i], dp[j] + x);
        }
        ans = std::max(ans, dp[i] + h[i]);
    }
        fout << std::fixed << std::setprecision(10);
    fout << ans << "\n";
        return 0;
}
