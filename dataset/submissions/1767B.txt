#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
        std::sort(a.begin() + 1, a.end());
    for (int i = 1; i < n; i++) {
        if (a[i] > a[0]) {
            a[0] = (a[0] + a[i] + 1) / 2;
        }
    }
        std::cout << a[0] << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    std::cin >> t;
        while (t--) {
        solve();
    }

}
