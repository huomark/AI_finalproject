#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> a(n), b(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
    for (int i = 0; i < n; i++) {
        std::cin >> b[i];
    }
    for (int i = 0; i < n; i++) {
        if (a[i] > b[i]) {
            std::cout << "NO\n";
            return;
        }
    }
    std::vector<int> l(n), r(n);
    std::vector<int> stk;
    for (int i = 0; i < n; i++) {
        while (!stk.empty() && a[stk.back()] <= a[i]) {
            stk.pop_back();
        }
        l[i] = stk.empty() ? -1 : stk.back();
        stk.push_back(i);
    }
    stk = {};
    for (int i = n - 1; i >= 0; i--) {
        while (!stk.empty() && a[stk.back()] <= a[i]) {
            stk.pop_back();
        }
        r[i] = stk.empty() ? n : stk.back();
        stk.push_back(i);
    }
        int j = 0;
    for (int i = 0; i < n; i++) {
        while (j < n && b[j] == a[i] && j > l[i] && j < r[i]) {
            j++;
        }
    }
    if (j == n) {
        std::cout << "YES\n";
    } else {
        std::cout << "NO\n";
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
