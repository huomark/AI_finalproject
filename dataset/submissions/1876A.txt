#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n, p;
    std::cin >> n >> p;
        std::vector<int> a(n), b(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
    for (int i = 0; i < n; i++) {
        std::cin >> b[i];
    }
        std::vector<int> ord(n);
    std::iota(ord.begin(), ord.end(), 0);
    std::sort(ord.begin(), ord.end(),
        [&](int i, int j) {
            return b[i] < b[j];
        });
        i64 ans = p;
    int k = n - 1;
    for (auto i : ord) {
        while (k > 0 && a[i] > 0) {
            a[i]--, k--;
            ans += std::min(p, b[i]);
        }
    }
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
