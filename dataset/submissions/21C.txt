#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n;
    std::cin >> n;
        std::vector<int> a(n), s(n + 1);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
        s[i + 1] = s[i] + a[i];
    }
        if (s[n] % 3 != 0) {
        std::cout << 0 << "\n";
        return 0;
    }
        i64 ans = 0;
    int cnt = 0;
    for (int i = 1; i < n; i++) {
        if (s[i] == s[n] / 3 * 2) {
            ans += cnt;
        }
        if (s[i] == s[n] / 3) {
            cnt += 1;
        }
    }
    std::cout << ans << "\n";
        return 0;
}
