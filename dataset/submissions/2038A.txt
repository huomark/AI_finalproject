#include <bits/stdc++.h>
 using i64 = long long;
using u64 = unsigned long long;
using u32 = unsigned;
using u128 = unsigned __int128;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n, k;
    std::cin >> n >> k;
        std::vector<int> a(n), b(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
    for (int i = 0; i < n; i++) {
        std::cin >> b[i];
    }
        i64 sum = 0;
    for (int i = 0; i < n; i++) {
        sum += a[i] / b[i];
    }
        if (sum < k) {
        for (int i = 1; i <= n; i++) {
            std::cout << 0 << " \n"[i == n];
        }
        return 0;
    }
        for (int i = 0; i < n; i++) {
        sum -= a[i] / b[i];
        int c = std::max(0LL, k - sum);
        k -= c;
        std::cout << c << " \n"[i == n - 1];
    }
        return 0;
}
