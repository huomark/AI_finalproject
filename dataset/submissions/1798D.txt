#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
        if (a == std::vector(n, 0)) {
        std::cout << "NO\n";
        return;
    }
        std::cout << "YES\n";
        i64 s = 0;
    int i = 0, j = n - 1;
    std::sort(a.begin(), a.end());
        std::vector<int> ans;
    while (i <= j) {
        if (s <= 0) {
            s += a[j];
            ans.push_back(a[j--]);
        } else {
            s += a[i];
            ans.push_back(a[i++]);
        }
    }
        for (int i = 0; i < n; i++) {
        std::cout << ans[i] << " \n"[i == n - 1];
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
