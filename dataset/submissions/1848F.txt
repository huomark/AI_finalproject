#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n;
    std::cin >> n;
        std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
    if (a == std::vector(n, 0)) {
        std::cout << 0 << "\n";
        return 0;
    }
        int ans = 1;
    for (int i = n / 2; i > 0; i /= 2) {
        std::vector<int> b(n);
        for (int j = 0; j < n; j++) {
            b[j] = a[j] ^ a[(i + j) % n];
        }
        if (b != std::vector(n, 0)) {
            a = std::move(b);
            ans += i;
        }
    }
    std::cout << ans << "\n";
        return 0;
}
