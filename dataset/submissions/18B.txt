#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n, d, m, l;
    std::cin >> n >> d >> m >> l;
        i64 cur = 0;
    while (cur < 1LL * m * n && cur % m <= l) {
        cur += (l - cur % m) / d * d;
        cur += d;
    }
    std::cout << cur << "\n";
        return 0;
}
