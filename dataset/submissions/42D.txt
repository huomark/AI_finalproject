#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n;
    std::cin >> n;
        std::vector<int> a(n);
    a[0] = 0;
    a[1] = 1;
    std::set<int> s{0, 1};
    for (int i = 2, x = 0; i < n; i++) {
        while (s.count(x)) {
            x++;
        }
        a[i] = x;
        for (int j = 0; j <= i; j++) {
            for (int k = 0; k <= i; k++) {
                if (j < i) {
                    s.insert(a[i] + a[j] - a[k]);
                }
                if (j < k) {
                    s.insert(a[j] + a[k] - a[i]);
                }
            }
        }
    }
        for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            std::cout << (i == j ? 0 : a[i] + a[j]) << " \n"[j == n-1];
        }
    }
        return 0;
}
