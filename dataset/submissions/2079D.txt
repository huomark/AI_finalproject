#include <bits/stdc++.h>
 using i64 = long long;
using u64 = unsigned long long;
using u32 = unsigned;
using u128 = unsigned __int128;
using i128 = __int128;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n, k;
    std::cin >> n >> k;
        std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
        std::priority_queue<int, std::vector<int>, std::greater<>> pq;
    i64 ans = 0;
    i64 sum = 0;
    for (int i = 0; i < n; i++) {
        if (pq.size() == k - 1) {
            ans = std::max(ans, sum + a[i] + i + 1);
        }
        pq.push(a[i]);
        sum += a[i];
        if (pq.size() >= k) {
            sum -= pq.top();
            pq.pop();
        }
    }
    std::cout << ans << "\n";
        return 0;
}
