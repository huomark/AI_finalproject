#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n, r1, r2;
    std::cin >> n >> r1 >> r2;
    r1--, r2--;
        std::vector<std::vector<int>> adj(n);
    for (int i = 0; i < n; i++) {
        if (i != r1) {
            int p;
            std::cin >> p;
            p--;
            adj[p].push_back(i);
            adj[i].push_back(p);
        }
    }
        std::vector<int> parent(n, -1);
    std::function<void(int)> dfs = [&](int x) {
        for (auto y : adj[x]) {
            if (y == parent[x]) {
                continue;
            }
            parent[y] = x;
            dfs(y);
        }
    };
    dfs(r2);
        for (int i = 0; i < n; i++) {
        if (i != r2) {
            std::cout << parent[i] + 1 << " ";
        }
    }
        return 0;
}
