#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n, k;
    std::cin >> n >> k;
        std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
        std::vector<int> p(n);
    std::iota(p.begin(), p.end(), 0);
    std::sort(p.begin(), p.end(),
        [&](int i, int j) {
            return a[i] < a[j];
        });
        int i = -1, j = -1, x = -1;
    int ans = 1 << k;
    for (int t = 1; t < n; t++) {
        if (ans > (a[p[t]] ^ a[p[t - 1]])) {
            ans = (a[p[t]] ^ a[p[t - 1]]);
            i = p[t - 1] + 1;
            j = p[t] + 1;
            x = ((1 << k) - 1) ^ (a[p[t]] | a[p[t - 1]]);
        }
    }
    std::cout << i << " " << j << " " << x << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
