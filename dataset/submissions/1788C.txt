#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        if (n % 2 == 0) {
        std::cout << "No\n";
        return;
    }
        std::cout << "Yes\n";
    for (int i = 1; i <= (n + 1) / 2; i++) {
        std::cout << i << " " << 2 * n - 2 * i + 2 << "\n";
    }
    for (int i = n; i > (n + 1) / 2; i--) {
        std::cout << i << " " << 3 * n - 2 * i + 2 << "\n";
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
