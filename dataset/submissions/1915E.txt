#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<i64> s(n + 1);
    std::set<i64> set{0LL};
    for (int i = 0; i < n; i++) {
        int a;
        std::cin >> a;
        s[i + 1] = s[i] + (i % 2 ? 1 : -1) * a;
        set.insert(s[i + 1]);
    }
    if (set.size() != n + 1) {
        std::cout << "YES\n";
    } else {
        std::cout << "NO\n";
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
