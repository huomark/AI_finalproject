#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    i64 n;
    std::cin >> n;
        std::string s;
    while (n) {
        int x = n % 9;
        s += '0' + (x < 4 ? x : x + 1);
        n /= 9;
    }
    std::reverse(s.begin(), s.end());
        std::cout << s << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
