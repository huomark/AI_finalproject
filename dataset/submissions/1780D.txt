#include <bits/stdc++.h>
 using i64 = long long;
 int query(int x) {
    std::cout << "- " << x << std::endl;
    int ans;
    std::cin >> ans;
    return ans;
}
 void solve() {
    int cur;
    std::cin >> cur;
        int k = 1, ans = 0;
    bool last = false;
    while (cur > 0) {
        ans += last ? k / 2 : k;
        int r = query(last ? k / 2 : k);
        if (r < cur) {
            last = false;
            cur = r;
        } else {
            last = true;
        }
        k *= 2;
    }
        std::cout << "! " << ans << std::endl;
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
