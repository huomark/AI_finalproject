#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n, x, p;
    std::cin >> n >> x >> p;
        for (int i = 1; i <= std::min(p, 2 * n); i++) {
        if ((x + 1LL * i * (i + 1) / 2) % n == 0) {
            std::cout << "Yes\n";
            return;
        }
    }
    std::cout << "No\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
