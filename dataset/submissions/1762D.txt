#include <bits/stdc++.h>
 using i64 = long long;
 int query(int x, int y) {
    std::cout << "? " << x + 1 << " " << y + 1 << std::endl;
    int res;
    std::cin >> res;
    return res;
}
 void solve() {
    int n;
    std::cin >> n;
        int x = 0, y = 1;
    int g = query(x, y);
    for (int i = 2; i < n; i++) {
        int gx = query(i, x);
        int gy = query(i, y);
        if (gx >= std::max(gy, g)) {
            y = i;
            g = gx;
        } else if (gy >= std::max(gx, g)) {
            x = i;
            g = gy;
        }
    }
        std::cout << "! " << x + 1 << " " << y + 1 << std::endl;
    int res;
    std::cin >> res;
    }
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
