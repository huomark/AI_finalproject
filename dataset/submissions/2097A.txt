 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
    std::sort(a.begin(), a.end());
    a.push_back(2E9);
        int state = 0;
    int cur = 0;
    int cnt = 0;
    for (auto x : a) {
        while (cur < x) {
            if (state == 0) {
                if (cnt >= 4) {
                    std::cout << "Yes\n";
                    return;
                }
                if (cnt >= 2) {
                    state = 1;
                }
            } else if (state == 1) {
                if (cnt >= 2) {
                    std::cout << "Yes\n";
                    return;
                }
                if (cnt == 0) {
                    state = 0;
                }
            }
            cnt = 0;
            cur++;
            if (state == 0) {
                cur = x;
            }
        }
        cnt++;
    }
    std::cout << "No\n";
}
