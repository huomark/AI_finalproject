#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n;
    std::cin >> n;
        std::vector<int> l(n), r(n);
    for (int i = 0; i < n; i++) {
        std::cin >> l[i] >> r[i];
    }
        std::vector<int> ord(n);
    std::iota(ord.begin(), ord.end(), 0);
    std::sort(ord.begin(), ord.end(), [&](int i, int j) {
        return l[i] < l[j];
    });
        std::vector<int> ans;
    for (int i = 0; i < n; i++) {
        int x = 0;
        bool ok = true;
        for (int j = 0; j < n; j++) {
            if (i != j) {
                if (l[ord[j]] < x) {
                    ok = false;
                    break;
                }
                x = r[ord[j]];
            }
        }
        if (ok) {
            ans.push_back(ord[i]);
        }
    }
    std::sort(ans.begin(), ans.end());
    std::cout << ans.size() << "\n";
    for (auto x : ans) {
        std::cout << x + 1 << " \n"[x == ans.back()];
    }
        return 0;
}
