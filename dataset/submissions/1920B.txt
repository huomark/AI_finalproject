#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n, k, x;
    std::cin >> n >> k >> x;
        std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
    std::sort(a.begin(), a.end());
        i64 ans = -1E18;
        std::vector<i64> sum(n + 1);
    for (int i = 0; i < n; i++) {
        sum[i + 1] = sum[i] + a[i];
    }
    for (int i = 0; i <= k; i++) {
        i64 res = -sum[n - i] + 2 * sum[std::max(0, n - i - x)];
        ans = std::max(ans, res);
    }
        std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
