#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        std::string s;
    std::cin >> s;
        int ans = 0;
    std::set<std::string> a;
    for (int i = 0; i < s.size(); i++) {
        for (int j = i; j < s.size(); j++) {
            auto t = s.substr(i, j - i + 1);
            if (a.count(t)) {
                ans = std::max(ans, int(t.size()));
            } else {
                a.insert(t);
            }
        }
    }
    std::cout << ans << "\n";
        return 0;
}
