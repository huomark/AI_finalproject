#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n;
    std::cin >> n;
        std::vector<std::pair<int, int>> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i].first >> a[i].second;
        if (a[i].first < a[i].second) {
            std::swap(a[i].first, a[i].second);
        }
    }
    std::sort(a.begin(), a.end());
        int last = -1E9;
    std::vector<int> ans;
        for (auto [r, l] : a) {
        if (l > last) {
            last = r;
            ans.push_back(r);
        }
    }
        std::cout << ans.size() << "\n";
    for (auto x : ans) {
        std::cout << x << " \n"[x == ans.back()];
    }
        return 0;
}
