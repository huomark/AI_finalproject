#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n, m, k;
    std::cin >> n >> m >> k;
        std::cout << std::fixed << std::setprecision(10);
    if (n - m < -k) {
        std::cout << 0. << "\n";
        return 0;
    }
        if (m <= k) {
        std::cout << 1. << "\n";
        return 0;
    }
        double ans = 1;
    for (int i = 1; i <= k + 1; i++) {
        ans *= m - i + 1;
        ans /= n + i;
    }
    ans = 1 - ans;
        std::cout << ans << "\n";
        return 0;
}
