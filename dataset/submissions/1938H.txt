#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
     int n;
    std::cin >> n;
     std::vector<int> a(n), b(n);
    for (int i = 0; i < n; i++) {
        std::string s;
        std::cin >> s;
         a[i] = std::count(s.begin(), s.end(), '0');
        b[i] = s.size() - a[i];
    }
     int ans = 0;
    int mva = 0, mvb = 0;
    for (int i = 0; i < n; i++) {
        if (a[i] < b[i]) {
            ans += a[i];
            mva++;
        } else {
            ans += b[i];
            mvb++;
        }
    }
     if (ans > 0 && (mva == n || mvb == n)) {
        int min = 1E9;
        for (int i = 0; i < n; i++) {
            min = std::min(min, std::abs(a[i] - b[i]));
        }
        ans += min;
    }
     std::cout << ans << "\n";
     return 0;
}
