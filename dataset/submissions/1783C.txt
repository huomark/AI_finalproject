#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n, m;
    std::cin >> n >> m;
        std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
        auto b = a;
    std::sort(b.begin(), b.end());
        std::vector<int> sum(n + 1);
    for (int i = 0; i < n; i++) {
        sum[i + 1] = sum[i] + b[i];
    }
        int ans = n + 1;
    for (int x = 1; x <= n; x++) {
        int res = n - x;
        if (m >= sum[x]) {
            ans = std::min(ans, n - x + 1);
        }
        if (x < n && m >= sum[x] + std::max(a[x] - b[x - 1], 0)) {
            ans = std::min(ans, std::max(1, n - x));
        }
    }
        std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
