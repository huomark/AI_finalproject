#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n, k;
    std::cin >> n >> k;
        std::string s;
    std::cin >> s;
        int cnt = std::count(s.begin(), s.end(), 'B');
    if (cnt == k) {
        std::cout << 0 << "\n";
        return;
    }
        for (int i = 0; i < n; i++) {
        cnt -= (s[i] == 'B');
        if (cnt == k) {
            std::cout << 1 << "\n";
            std::cout << i + 1 << " A\n";
            return;
        }
        if (cnt + i + 1 == k) {
            std::cout << 1 << "\n";
            std::cout << i + 1 << " B\n";
            return;
        }
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
