 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::string s[2];

        for (int t = 0; t < 2; t++) {
        int x = t;
        int ok = 1;
        for (int i = 0; i < n; i++) {
            if (s[x][i] != 'B') ok = 0;
            if (s[!x][i] == 'B') x ^= 1;
        }
                if (ok) {

        }
    }

}
