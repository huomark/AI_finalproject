#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n;
    std::cin >> n;
        std::vector<int> cnt(n);
    for (int i = 0; i < n; i++) {
        int t;
        std::cin >> t;
        cnt[t]++;
    }
        int ans = n;
    for (int i = 1; i < n; i++) {
        if (cnt[i] > 0) {
            ans--;
        }
    }
    std::cout << ans << "\n";
        return 0;
}
