#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n, k;
    std::cin >> n >> k;
        std::string s;
    std::cin >> s;
        int cnt[26] {};
    for (auto c : s) {
        cnt[c - 'a']++;
    }
        int ans = std::count_if(cnt, cnt + 26,
        [&](int x) {
            return x % 2 == 1;
        });
    if (k >= ans - 1) {
        std::cout << "YES\n";
    } else {
        std::cout << "NO\n";
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
