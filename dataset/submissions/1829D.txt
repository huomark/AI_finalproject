#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n, m;
    std::cin >> n >> m;
        int t = 0;
    while (n % 3 == 0 && m % 3 == 0) {
        n /= 3;
        m /= 3;
    }
    while (n % 3 == 0) {
        n /= 3;
        t++;
    }
    if (m % n != 0) {
        std::cout << "NO\n";
        return;
    }
    m /= n;
        if ((m & (m - 1)) == 0 && m <= (1 << t)) {
        std::cout << "YES\n";
    } else {
        std::cout << "NO\n";
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
