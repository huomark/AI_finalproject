#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n;
    std::cin >> n;
        std::vector<i64> dp(2 * n + 1, 1E18);
    dp[n] = 0;
        for (int i = 0; i < n; i++) {
        int t, c;
        std::cin >> t >> c;
                std::vector<i64> g(2 * n + 1, 1E18);
        for (int j = 0; j <= 2 * n; j++) {
            g[std::min(2 * n, j + t)] = std::min(g[std::min(2 * n, j + t)], dp[j] + c);
            if (j) {
                g[j - 1] = std::min(g[j - 1], dp[j]);
            }
        }
        dp = g;
    }
        i64 ans = *std::min_element(dp.begin() + n, dp.end());
    std::cout << ans << "\n";
        return 0;
}
