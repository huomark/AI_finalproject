#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n, m;
    std::cin >> n >> m;
        i64 ans = 0;
        std::vector<std::string> s(m);
    std::vector<int> k(m);
    for (int i = 0; i < m; i++) {
        std::cin >> s[i] >> k[i];
    }
        std::function<void(int, std::vector<int>)> dfs = [&](int x, std::vector<int> a) {
        if (x == n) {
            ans += a == k;
            return;
        }
        for (int i = 0; i < m; i++) {
            if (a[i] > k[i]) {
                return;
            }
        }
        for (int c = 0; c < 2; c++) {
            auto b = a;
            for (int i = 0; i < m; i++) {
                b[i] += s[i][x] == '0' + c;
            }
            dfs(x+1, b);
        }
    };
    dfs(0, std::vector(m, 0));
        std::cout << ans << "\n";
        return 0;
}
