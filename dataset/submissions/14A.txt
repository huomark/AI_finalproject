#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n, m;
    std::cin >> n >> m;
        std::vector<std::string> s(n);
    for (int i = 0; i < n; i++) {
        std::cin >> s[i];
    }
        int minx = n, maxx = -1, miny = m, maxy = -1;
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            if (s[i][j] == '*') {
                minx = std::min(minx, i);
                maxx = std::max(maxx, i);
                miny = std::min(miny, j);
                maxy = std::max(maxy, j);
            }
        }
    }
        for (int i = minx; i <= maxx; i++) {
        for (int j = miny; j <= maxy; j++) {
            std::cout << s[i][j];
        }
        std::cout << "\n";
    }
        return 0;
}
