#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int a, b;
    std::cin >> a >> b;
        int ans = 2E9;
        for (int i = 1; i <= 1E5; i++) {
        ans = std::min(ans, i + 1 + (a - 1) / i + (b - 1) / i);
    }
        std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
