#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
        int lst = a[n - 1];
    i64 ans = 0;
    for (int i = n - 2; i >= 0; i--) {
        int t = (a[i] - 1) / lst + 1;
        ans += t - 1;
        lst = a[i] / t;
    }
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
