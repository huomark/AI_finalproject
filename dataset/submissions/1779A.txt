#include <bits/stdc++.h>
 void solve() {
    int n;
    std::cin >> n;
        std::string s;
    std::cin >> s;
        int pos = -1;
    for (int i = 1; i < n; i++) {
        if (s[i - 1] == 'R' && s[i] == 'L') {
            std::cout << 0 << "\n";
            return;
        }
        if (s[i - 1] == 'L' && s[i] == 'R') {
            pos = i;
        }
    }
    std::cout << pos << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
