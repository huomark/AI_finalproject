#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n;
    std::cin >> n;
        std::string s;
    std::cin >> s;
        std::vector<int> a(n - 1);
    for (int i = 0; i < n - 1; i++) {
        a[i] = s[i] ^ s[i + 1];
    }
        std::vector f(20, std::vector<int>(n));
    for (int i = 0; i < 20; i++) {
        for (int j = 0; j < n; j++) {
            f[i][j] = a[j];
            if (j >= (1 << i)) {
                f[i][j] += f[i][j - (1 << i)];
            }
        }
    }
        int q;
    std::cin >> q;
        while (q--) {
        int l, r;
        std::cin >> l >> r;
                int ans = 0;
        int k = std::__lg(r - l + 1);
        for (int i = 1; i <= k; i++) {
            int a = f[i][r - (1 << (i - 1))] - (l >= (1 << (i - 1)) + 1 ? f[i][l - (1 << (i - 1)) - 1] : 0);
            ans += std::min(a, (1 << (k - i)) - a);
        }
        ans = (ans + 1) / 2;
        std::cout << ans << "\n";
    }
        return 0;
}
