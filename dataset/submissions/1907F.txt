#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
        int ans = -1;
    int f = 0;
    int p = -1;
    for (int i = 0; i < n; i++) {
        if (a[i] > a[(i + 1) % n]) {
            f += 1;
            p = i + 1;
        }
    }
    if (f == 0) {
        ans = 0;
    } else if (f == 1) {
        int res = std::min(p + 2, n - p);
        if (ans == -1 || ans > res) {
            ans = res;
        }
    }
    f = 0;
    for (int i = 0; i < n; i++) {
        if (a[i] < a[(i + 1) % n]) {
            f += 1;
            p = i + 1;
        }
    }
    if (f == 0) {
        ans = 0;
    } else if (f == 1) {
        int res = std::min(p + 1, n - p + 1);
        if (ans == -1 || ans > res) {
            ans = res;
        }
    }
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
