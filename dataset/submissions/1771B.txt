#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n, m;
    std::cin >> n >> m;
        std::vector<int> r(n, n - 1);
    for (int i = 0; i < m; i++) {
        int x, y;
        std::cin >> x >> y;
        x--, y--;
        if (x > y) {
            std::swap(x, y);
        }
        r[x] = std::min(r[x], y - 1);
    }
        i64 ans = 0;
    for (int i = n - 1; i >= 0; i--) {
        if (i < n - 1) {
            r[i] = std::min(r[i], r[i + 1]);
        }
        ans += r[i] - i + 1;
    }
        std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
