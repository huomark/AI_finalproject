#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n;
    std::cin >> n;
        std::map<int, int> f;
    for (int i = 0; i < n; i++) {
        int x, d;
        std::cin >> x >> d;
        f[x] = d;
        if (f.count(x + d) && f[x + d] == -d) {
            std::cout << "YES\n";
            return 0;
        }
    }
    std::cout << "NO\n";
        return 0;
}
