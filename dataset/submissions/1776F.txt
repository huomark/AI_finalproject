#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n, m;
    std::cin >> n >> m;
        std::vector<int> u(m), v(m), deg(n);
    for (int i = 0; i < m; i++) {
        std::cin >> u[i] >> v[i];
        u[i]--, v[i]--;
        deg[u[i]]++, deg[v[i]]++;
    }
        for (int x = 0; x < n; x++) {
        if (deg[x] < n - 1) {
            std::cout << 2 << "\n";
            for (int i = 0; i < m; i++) {
                std::cout << ((u[i] == x || v[i] == x) ? 1 : 2) << " \n"[i == m - 1];
            }
            return;
        }
    }
        std::cout << 3 << "\n";
    for (int i = 0; i < m; i++) {
        int x;
        if (u[i] + v[i] == 1) {
            x = 1;
        } else if (u[i] == 0 || v[i] == 0) {
            x = 2;
        } else {
            x = 3;
        }
        std::cout << x << " \n"[i == m - 1];
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
