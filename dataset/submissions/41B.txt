#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n, b;
    std::cin >> n >> b;
        std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
    int ans = b;
    for (int i = 0; i < n; i++) {
        for (int j = i+1; j < n; j++) {
            ans = std::max(ans, b + (a[j] - a[i]) * (b / a[i]));
        }
    }
    std::cout << ans << "\n";
        return 0;
}
