#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n, x;
    std::cin >> n >> x;
        std::vector<int> cnt(x + 1);
    for (int i = 0; i < n; i++) {
        int a;
        std::cin >> a;
        cnt[std::min(a, x)]++;
    }
        for (int i = 1; i < x; i++) {
        if (cnt[i] % (i + 1)) {
            std::cout << "No\n";
            return 0;
        }
        cnt[i + 1] += cnt[i] / (i + 1);
    }
    std::cout << "Yes\n";
        return 0;
}
