#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n, H;
    std::cin >> n >> H;
        std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
        std::nth_element(a.begin(), a.begin() + 1, a.end(), std::greater());
    int ans = H / (a[0] + a[1]) * 2;
    H %= a[0] + a[1];
    if (H > 0) {
        ans += 1;
    }
    if (H > a[0]) {
        ans += 1;
    }
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
