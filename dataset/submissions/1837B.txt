#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
    std::string s;
    std::cin >> s;
        int ans = 0;
    for (int i = 0, j = 0; i < n; i = j) {
        while (j < n && s[i] == s[j]) {
            j++;
        }
        ans = std::max(ans, j - i);
    }
    std::cout << ans + 1 << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
