#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
        if (a.back() != 0) {
        std::cout << "NO\n";
        return;
    }
        std::cout << "YES\n";
        std::vector<int> ans;
        for (int i = 0, j = -1; i < n; i++) {
        if (a[i] == 0) {
            ans.push_back(i - j - 1);
            for (int k = j + 1; k < i; k++) {
                ans.push_back(0);
            }
            j = i;
        }
    }
    std::reverse(ans.begin(), ans.end());
        for (int i = 0; i < n; i++) {
        std::cout << ans[i] << " \n"[i == n - 1];
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
