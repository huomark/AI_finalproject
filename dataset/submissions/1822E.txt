#include <bits/stdc++.h> 
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::string s;
    std::cin >> s;
        std::array<int, 26> cnt{};
    for (int i = 0; i < n; i++) {
        cnt[s[i] - 'a']++;
    }
    if (n % 2 || *std::max_element(cnt.begin(), cnt.end()) > n/2) {
        std::cout << -1 << "\n";
        return;
    }
        cnt = {};
    int tot = 0;
    for (int i = 0; i < n/2; i++) {
        if (s[i] == s[n-1-i]) {
            cnt[s[i] - 'a']++;
            tot++;
        }
    }
        int ans = std::max((tot+1) / 2, *std::max_element(cnt.begin(), cnt.end()));
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
