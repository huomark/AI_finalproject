#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int A;
    std::cin >> A;
        int num = 0, den = 0;
    for (int i = 2; i < A; i++) {
        int x = A;
        while (x > 0) {
            num += x % i;
            x /= i;
        }
        den += 1;
    }
        int g = std::gcd(num, den);
    num /= g;
    den /= g;
    std::cout << num << "/" << den << "\n";
        return 0;
}
