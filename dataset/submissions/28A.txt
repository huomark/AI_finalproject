#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n, m;
    std::cin >> n >> m;
        std::vector<int> x(n), y(n);
    for (int i = 0; i < n; i++) {
        std::cin >> x[i] >> y[i];
    }
        std::set<std::pair<int, int>> f;
    for (int i = 0; i < m; i++) {
        int x;
        std::cin >> x;
        f.emplace(x, i);
    }
        for (int t = 0; t < 2; t++) {
        auto g = f;
        bool ok = true;
        std::vector<int> ans(n, -1);
        for (int i = t; i < n; i += 2) {
            int len = std::abs(x[(i - 1 + n) % n] - x[(i + 1) % n])
                + std::abs(y[(i - 1 + n) % n] - y[(i + 1) % n]);
            auto it = g.lower_bound({len, 0});
            if (it != g.end() && it->first == len) {
                ans[i] = it->second + 1;
                g.erase(it);
            } else {
                ok = false;
                break;
            }
        }
        if (ok) {
            std::cout << "YES\n";
            for (int i = 0; i < n; i++) {
                std::cout << ans[i] << " \n"[i == n - 1];
            }
            return 0;
        }
    }
        std::cout << "NO\n";
        return 0;
}
