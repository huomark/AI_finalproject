#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int q;
    std::cin >> q;
        std::map<i64, i64> f;
    std::set<i64> s{0};
        for (int i = 0; i < q; i++) {
        char o;
        i64 x;
        std::cin >> o >> x;
        if (o == '+') {
            s.insert(x);
        } else {
            while (s.contains(f[x])) {
                f[x] += x;
            }
            std::cout << f[x] << "\n";
        }
    }
        return 0;
}
