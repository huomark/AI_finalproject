#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<i64> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
        i64 k = 1;
    while (true) {
        for (int i = 0; i < n; i++) {
            if (a[i] % k != a[0] % k) {
                std::cout << k << "\n";
                return;
            }
        }
        k *= 2;
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
