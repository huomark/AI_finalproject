#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int b, n;
    std::cin >> b >> n;
        int p = 1;
    for (int k = 1; k <= n; k++) {
        p = 1LL * p * b % n;
        if (p == 0) {
            std::cout << 1 << " " << k << "\n";
            return;
        }
        if (p == 1) {
            std::cout << 2 << " " << k << "\n";
            return;
        }
        if (p == n - 1) {
            std::cout << 3 << " " << k << "\n";
            return;
        }
    }
    std::cout << 0 << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
