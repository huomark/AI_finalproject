#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int x;
    std::cin >> x;
        x = std::abs(x);
        int ans = 0;
    while (true) {
        i64 s = 1LL * ans * (ans + 1) / 2;
        if (s >= x && (s - x) % 2 == 0) {
            break;
        }
        ans += 1;
    }
    std::cout << ans << "\n";
        return 0;
}
