#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int q;
    std::cin >> q;
        std::vector<int> a;
    int bad = 0;
    for (int i = 0; i < q; i++) {
        int x;
        std::cin >> x;
                if (a.empty()) {
            a.push_back(x);
            std::cout << 1;
        } else {
            int nbad = bad - (a.back() > a[0]) + (a.back() > x) + (x > a[0]);
            if (nbad <= 1) {
                bad = nbad;
                a.push_back(x);
                std::cout << 1;
            } else {
                std::cout << 0;
            }
        }
    }
        std::cout << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
