#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n, k;
    std::cin >> n >> k;
        std::vector<int> a(n);
    int cur = 0;
    for (int x = 0; x < k; x++) {
        if (x % 2 == 0) {
            for (int i = x; i < n; i += k) {
                a[i] = ++cur;
            }
        } else {
            for (int i = (n - 1) - (n - 1 - x) % k; i >= 0; i -= k) {
                a[i] = ++cur;
            }
        }
    }
        for (int i = 0; i < n; i++) {
        std::cout << a[i] << " \n"[i == n - 1];
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
