#include <bits/stdc++.h>
 using i64 = long long;
using u64 = unsigned long long;

using i128 = __int128;
 void solve() {
    int n, k, p;
    std::cin >> n >> k >> p;
        k = std::abs(k);
    if (k > n * p) {
        std::cout << -1 << "\n";
    } else {
        std::cout << (k + p - 1) / p << "\n";
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    std::cin >> t;

    }

}
