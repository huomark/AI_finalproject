#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
        std::vector<int> nxt(n);
    nxt[n - 1] = n;
    for (int i = n - 2; i >= 0; i--) {
        nxt[i] = a[i] == a[i + 1] ? nxt[i + 1] : i + 1;
    }
        int q;
    std::cin >> q;
        while (q--) {
        int l, r;
        std::cin >> l >> r;
        l--;
        if (nxt[l] < r) {
            std::cout << l + 1 << " " << nxt[l] + 1 << "\n";
        } else {
            std::cout << -1 << " " << -1 << "\n";
        }
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
