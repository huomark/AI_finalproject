#include <bits/stdc++.h>
 using i64 = long long;
using u64 = unsigned long long;
using u32 = unsigned;
using u128 = unsigned __int128;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> p(n);
    for (int i = 0; i < n; i++) {
        std::cin >> p[i];
        p[i]--;
    }
        std::vector<int> bel(n, -1), siz(n);
    for (int x = 0; x < n; x++) {
        if (bel[x] != -1) {
            continue;
        }
        for (int i = x; bel[i] == -1; i = p[i]) {
            bel[i] = x;
            siz[x]++;
        }
    }
        int ans = 0;
    for (int i = 0; i < n; i++) {
        int d;
        std::cin >> d;
        d--;
        ans += siz[bel[d]];
        siz[bel[d]] = 0;
        std::cout << ans << " \n"[i == n - 1];
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
