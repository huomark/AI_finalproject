#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n, c;
    std::cin >> n >> c;
        std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
        a[i] += i + 1;
    }
        std::sort(a.begin(), a.end());
        int ans = 0;
    for (auto x : a) {
        if (x <= c) {
            ans++;
            c -= x;
        }
    }
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
