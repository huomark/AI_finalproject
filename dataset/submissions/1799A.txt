#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n, m;
    std::cin >> n >> m;
        std::vector<int> p(m);
    std::vector<int> cnt(m);
    std::vector<int> ans(n, -1);
    int x = n - 1;
    for (int i = 0; i < m; i++) {
        std::cin >> p[i];
        p[i]--;
        if (cnt[p[i] - n]++ == 0) {
            if (x >= 0) {
                ans[x] = i + 1;
            }
            x--;
        }
    }
    for (int i = 0; i < n; i++) {
        std::cout << ans[i] << " \n"[i == n - 1];
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
