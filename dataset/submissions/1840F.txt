#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n, m, r;
    std::cin >> n >> m >> r;
        std::vector<std::array<int, 3>> shot(r);
    std::vector<int> tm;
    for (int i = 0; i < r; i++) {
        int t, d, p;
        std::cin >> t >> d >> p;
        shot[i] = {t, d, p};
        tm.push_back(t);
    }
    std::sort(tm.begin(), tm.end());
        tm.erase(std::unique(tm.begin(), tm.end()), tm.end());
    r = tm.size();
        std::vector sh(n + 1, std::vector(m + 1, std::vector(r + 1, false)));
    for (auto [t, d, p] : shot) {
        t = std::lower_bound(tm.begin(), tm.end(), t) - tm.begin();
        if (d == 1) {
            for (int j = 0; j <= m; j++) {
                sh[p][j][t] = true;
            }
        } else {
            for (int i = 0; i <= n; i++) {
                sh[i][p][t] = true;
            }
        }
    }
        std::vector dp(n + 1, std::vector(m + 1, std::vector(r + 1, -1)));
    dp[0][0][0] = 0;
        for (int i = 0; i <= n; i++) {
        for (int j = 0; j <= m; j++) {
            if (i > 0) {
                for (int k = 0; k <= r; k++) {
                    if (dp[i - 1][j][k] != -1) {
                        if (dp[i][j][k] == -1 || dp[i][j][k] > dp[i - 1][j][k] + 1) {
                            dp[i][j][k] = dp[i - 1][j][k] + 1;
                        }
                        if (k < r && !sh[i][j][k]) {
                            dp[i][j][k + 1] = tm[k];
                        }
                    }
                }
            }
            if (j > 0) {
                for (int k = 0; k <= r; k++) {
                    if (dp[i][j - 1][k] != -1) {
                        if (dp[i][j][k] == -1 || dp[i][j][k] > dp[i][j - 1][k] + 1) {
                            dp[i][j][k] = dp[i][j - 1][k] + 1;
                        }
                        if (k < r && !sh[i][j][k]) {
                            dp[i][j][k + 1] = tm[k];
                        }
                    }
                }
            }
            for (int k = 0; k < r; k++) {
                if (dp[i][j][k] >= tm[k]) {
                    dp[i][j][k] = -1;
                }
            }
            for (int k = 0; k < r; k++) {
                if (dp[i][j][k] != -1 && !sh[i][j][k]) {
                    dp[i][j][k + 1] = tm[k];
                }
            }
        }
    }
        int ans = -1;
    for (int k = 0; k <= r; k++) {
        if (dp[n][m][k] != -1) {
            ans = dp[n][m][k];
            break;
        }
    }
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
     return 0;
}
