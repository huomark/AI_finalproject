#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> a(n), b(n), inva(n), invb(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
        a[i]--;
        inva[a[i]] = i;
    }
    for (int i = 0; i < n; i++) {
        std::cin >> b[i];
        b[i]--;
        invb[b[i]] = i;
    }
        int ans = 0;
    for (int i = n - 1; i >= 0; i--) {
        if (i < n - 1 && inva[b[i]] > inva[b[i + 1]]) {
            ans = i + 1;
            break;
        }
    }
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
