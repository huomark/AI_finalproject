#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n, m;
    std::cin >> n >> m;
        std::vector<int> f(m + 1);
    std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
        if (a[i] <= m) {
            f[a[i]] += 1;
        }
    }
    for (int i = m; i; i--) {
        for (int j = 2 * i; j <= m; j += i) {
            f[j] += f[i];
        }
    }
        int l = std::max_element(f.begin(), f.end()) - f.begin();
    if (l == 0) {
        std::cout << 1 << " " << 0 << "\n";
        return 0;
    }
    int k = f[l];
    std::cout << l << " " << k << "\n";
    for (int i = 0; i < n; i++) {
        if (l % a[i] == 0) {
            std::cout << i + 1 << " ";
        }
    }
    std::cout << "\n";
        return 0;
}
