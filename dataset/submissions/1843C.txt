#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    i64 ans = 0;
    i64 n;
    std::cin >> n;
        while (n) {
        ans += n;
        n /= 2;
    }
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
