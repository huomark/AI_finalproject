#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n, s;
    std::cin >> n >> s;
        std::vector<std::vector<std::pair<int, int>>> a(n);
    for (int i = 0; i < n; i++) {
        int k;
        std::cin >> k;
        a[i].resize(k);
        for (int j = 0; j < k; j++) {
            int x, y;
            std::cin >> x >> y;
            a[i][j] = {x, y};
        }
    }
        int ans = 0;
    for (int x = 0; x < n; x++) {
        for (int y = x+1; y < n; y++) {
            auto A = a[x], B = a[y];
            int i = 0, j = 0;
            int lead = A[0] > B[0];
            int d = 0;
                        while (i < A.size() && j < B.size()) {
                int t = std::min(A[i].second, B[j].second);
                d += (A[i].first - B[j].first) * t;
                A[i].second -= t;
                B[j].second -= t;
                if (!A[i].second) {
                    i++;
                }
                if (!B[j].second) {
                    j++;
                }
                if (d && (d > 0) != lead) {
                    ans += 1;
                    lead ^= 1;
                }
            }
        }
    }
    std::cout << ans << "\n";
        return 0;
}
