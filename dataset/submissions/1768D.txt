#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> p(n);
    for (int i = 0; i < n; i++) {
        std::cin >> p[i];
        p[i]--;
    }
        std::vector<int> vis(n, -1);
    int ans = n;
    int sameAdj = 0;
    for (int i = 0; i < n; i++) {
        if (vis[i] != -1) continue;
                ans--;
        int j = i;
        while (vis[j] == -1) {
            vis[j] = i;
            j = p[j];
        }
    }
        for (int i = 1; i < n; i++) {
        if (vis[i] == vis[i - 1]) {
            sameAdj = 1;
        }
    }
        std::cout << ans + (sameAdj ? -1 : 1) << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
