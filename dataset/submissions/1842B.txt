#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n, x;
    std::cin >> n >> x;
        int ans = 0;
    for (int i = 0; i < 3; i++) {
        int u = 0;
        bool ok = true;
        for (int j = 0; j < n; j++) {
            int a;
            std::cin >> a;
            if ((a & x) != a) {
                ok = false;
            }
            if (ok) {
                u |= a;
            }
        }
        ans |= u;
    }
    if (ans == x) {
        std::cout << "Yes\n";
    } else {
        std::cout << "No\n";
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
