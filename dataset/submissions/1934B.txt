#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        int ans = n;
    for (int a = 0; a < 3; a++) {
        for (int b = 0; b < 2; b++) {
            for (int c = 0; c < 5; c++) {
                for (int d = 0; d < 3; d++) {
                    int res = n - a - 3 * b - 6 * c - 10 * d;
                    if (res >= 0 && res % 15 == 0) {
                        ans = std::min(ans, res / 15 + a + b + c + d);
                    }
                }
            }
        }
    }
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
