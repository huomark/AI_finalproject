#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    std::string s;
    std::cin >> s;
        int ans = 0;
    int n = s.size();
    for (int i = 0, j = -1; i <= n; i++) {
        if (i == n || s[i] == 'C') {
            int min = n;
            for (int x = j + 1, y = j; x <= i; x++) {
                if (x < i && s[x] == 'A') {
                    ans++;
                } else {
                    min = std::min(min, x - y - 1);
                    y = x;
                }
            }
            ans -= min;
        }
    }
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
