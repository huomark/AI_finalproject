#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::string s;
    std::cin >> s;
        int ans = n + 1;
    for (char x = 'a'; x <= 'z'; x++) {
        int l = 0, r = n - 1;
        int res = 0;
        while (l < r) {
            if (s[l] == s[r]) {
                l++, r--;
            } else if (s[l] == x) {
                l++, res++;
            } else if (s[r] == x) {
                r--, res++;
            } else {
                res = n + 1;
                break;
            }
        }
        ans = std::min(ans, res);
    }
    if (ans == n + 1) {
        ans = -1;
    }
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
