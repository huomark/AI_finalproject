#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n;
    std::cin >> n;
        std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
        std::vector<int> p;
    int ans = 1;
    for (int i = 0, j = 0; i < n; i++) {
        if (a[i] != -1) {
            if (p.size() < 2) {
                p.push_back(i);
            } else if (a[i] != a[p[0]] + (a[p[1]] - a[p[0]]) / (p[1] - p[0]) * 1LL * (i - p[0])) {
                ans++;
                j = i;
                p = {i};
            }
        }
        if (p.size() == 2) {
            if ((a[p[1]] - a[p[0]]) % (p[1] - p[0]) != 0
                || a[p[0]] + (a[p[1]] - a[p[0]]) / (p[1] - p[0]) * 1LL * (j - p[0]) <= 0
                || a[p[0]] + (a[p[1]] - a[p[0]]) / (p[1] - p[0]) * 1LL * (i - p[0]) <= 0) {
                ans++;
                j = i;
                p.clear();
                if (a[i] != -1) {
                    p.push_back(i);
                }
            }
        }
    }
    std::cout << ans << "\n";
        return 0;
}
