#include <bits/stdc++.h>
 using i64 = long long;
using u64 = unsigned long long;
using u32 = unsigned;
using u128 = unsigned __int128;
 void solve() {
    int k, l1, r1, l2, r2;
    std::cin >> k >> l1 >> r1 >> l2 >> r2;
        int ans = 0;
    for (i64 p = 1; p <= r2; p *= k) {
        int r = r2 / p;
        int l = (l2 - 1) / p + 1;
        ans += std::max(0, std::min(r1, r) - std::max(l1, l) + 1);
    }
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
