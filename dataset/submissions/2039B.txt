 using i64 = long long;

 void solve() {
    std::string s;
    std::cin >> s;
        for (int i = 0; i + 2 <= s.size(); i++) {
        if (s[i] == s[i + 1]) {
            std::cout << s.substr(i, 2) << "\n";
            return;
        }
    }
    for (int i = 0; i + 3 <= s.size(); i++) {
        if (s[i] != s[i + 2]) {
            std::cout << s.substr(i, 3) << "\n";
            return;
        }
    }
    std::cout << -1 << "\n";
}
 int main() {

    std::cin.tie(nullptr);

}
