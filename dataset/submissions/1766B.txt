#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::string s;
    std::cin >> s;
        std::set<int> t;
    int ok = 0;
    for (int i = 0; i < n - 1; i++) {
        if (t.contains(s[i] << 8 | s[i + 1])) ok = 1;
        if (i) t.insert(s[i - 1] << 8 | s[i]);
    }
        if (ok) std::cout << "YES\n";
    else std::cout << "NO\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) solve();
        return 0;
}
