#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n, m, k;
    std::cin >> n >> m >> k;
        std::vector<int> a(n), b(m);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
    for (int i = 0; i < m; i++) {
        std::cin >> b[i];
    }
    std::sort(a.begin(), a.end());
    std::sort(b.begin(), b.end());
        std::array<int, 4> cnt{};
    for (int x = 1; x <= k; x++) {
        int va = std::binary_search(a.begin(), a.end(), x);
        int vb = std::binary_search(b.begin(), b.end(), x);
        cnt[va * 2 + vb]++;
    }
        if (cnt[0] == 0 && cnt[1] + cnt[3] >= k / 2 && cnt[2] + cnt[3] >= k / 2) {
        std::cout << "YES\n";
    } else {
        std::cout << "NO\n";
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
