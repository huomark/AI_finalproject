#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::string a;
    std::cin >> a;
    std::vector<int> pre(n + 1);
    for (int i = 0; i < n; i++) {
        pre[i + 1] = pre[i] + a[i] - '0';
    }
        int i = -1;
    for (int j = 0; j <= n; j++) {
        int left = j - pre[j];
        int right = pre[n] - pre[j];
        if (left * 2 >= j && right * 2 >= n - j) {
            if (i == -1 || std::abs(n - 2 * j) < std::abs(n - 2 * i)) {
                i = j;
            }
        }
    }
    std::cout << i << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
