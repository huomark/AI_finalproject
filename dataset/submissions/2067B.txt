#include <bits/stdc++.h>
 using i64 = long long;
using u64 = unsigned long long;
using u32 = unsigned;
using u128 = unsigned __int128;
using i128 = __int128;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> cnt(2 * n);
    for (int i = 0; i < n; i++) {
        int a;
        std::cin >> a;
        a--;
        cnt[a]++;
    }
        for (int i = 0; i < 2 * n; i++) {
        if (cnt[i] == 1) {
            std::cout << "No\n";
            return;
        }
        if (cnt[i] > 2) {
            cnt[i + 1] += cnt[i] - 2;
        }
    }
        std::cout << "Yes\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
