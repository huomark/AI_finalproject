#include <bits/stdc++.h>
 using ll = long long;
using namespace std;
#define For(i,l,r) for(int i=(int)(l);i<=(int)(r);i++)
const int N=500005;
int n,a[N];
int get(ll x){
    if(x<0)return -100;
    return x?get(x>>1)+(x&1):0;
}
int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
     int t;
    cin>>t;
    while(t--){
        cin>>n;
        int mx=-2;
        For(i,1,n)cin>>a[i];
        For(i,1,n-1)mx=max(mx,a[i]-a[i+1]);
        ll fi=0;
        For(i,1,n-1)if(a[i]-a[i+1]==mx){
            fi=((ll)1<<(mx+1))-i; 
            //cerr<<mx<<" "<<fi<<endl;
            break;
        }
        mx++;
        int mj=a[1]-get(fi);
        For(i,1,mj)fi+=(ll)1<<++mx;
        //cerr<<fi<<endl;
        For(i,1,n)if(get(fi+i-1)!=a[i]){
            fi=-1; break;
        }
        cout<<fi<<"\n";
    }
     return 0;
}
