#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n, k, x;
    std::cin >> n >> k >> x;
        if (k > n || k > x + 1) {
        std::cout << -1 << "\n";
    } else if (k == x) {
        std::cout << k * (k - 1) / 2 + (n - k) * (k - 1) << "\n";
    } else {
        std::cout << k * (k - 1) / 2 + (n - k) * x << "\n";
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
