#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n, k;
    std::cin >> n >> k;
        std::string s;
    std::cin >> s;
        std::vector<int> l(k), r(k);
    for (int i = 0; i < k; i++) {
        std::cin >> l[i];
        l[i]--;
    }
    for (int i = 0; i < k; i++) {
        std::cin >> r[i];
        r[i]--;
    }
        int q;
    std::cin >> q;
        std::vector<int> f(n);
        while (q--) {
        int x;
        std::cin >> x;
        x--;
        f[x] ^= 1;
    }
        for (int i = 0; i < k; i++) {
        int rev = 0;
        for (int j = l[i]; j <= l[i] + r[i] - j; j++) {
            rev ^= f[j] ^ f[l[i] + r[i] - j];
            if (rev) {
                std::swap(s[j], s[l[i] + r[i] - j]);
            }
        }
    }
    std::cout << s << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
