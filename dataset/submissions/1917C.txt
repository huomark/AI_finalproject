#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n, k, d;
    std::cin >> n >> k >> d;
        std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
        std::vector<int> v(k);
    for (int i = 0; i < k; i++) {
        std::cin >> v[i];
    }
        int ans = 0;
    for (int i = 0; i < d && i < 2 * n + 1; i++) {
        int res = 0;
        for (int j = 0; j < n; j++) {
            res += (a[j] == j + 1);
        }
        ans = std::max(ans, res + (d - 1 - i) / 2);
        int b = v[i % k];
        for (int j = 0; j < b; j++) {
            a[j] += 1;
        }
    }
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
