#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n;
    std::cin >> n;
        std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
        i64 sum = std::accumulate(a.begin(), a.end(), 0LL);
        i64 s = 0;
    int ans = 0;
    for (int i = 0; i < n - 1; i++) {
        s += a[i];
        if (s * 2 == sum) {
            ans += 1;
        }
    }
    std::cout << ans << "\n";
        return 0;
}
