#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> a(n), b(n), c(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
    for (int i = 0; i < n; i++) {
        std::cin >> b[i];
    }
    for (int i = 0; i < n; i++) {
        std::cin >> c[i];
    }
        std::vector<int> pa(n), pb(n), pc(n);
    std::iota(pa.begin(), pa.end(), 0);
    std::iota(pb.begin(), pb.end(), 0);
    std::iota(pc.begin(), pc.end(), 0);
    std::sort(pa.begin(), pa.end(),
        [&](int i, int j) {
            return a[i] > a[j];
        });
    std::sort(pb.begin(), pb.end(),
        [&](int i, int j) {
            return b[i] > b[j];
        });
    std::sort(pc.begin(), pc.end(),
        [&](int i, int j) {
            return c[i] > c[j];
        });
    int ans = 0;
    for (int i = 0; i < 3; i++) {
        for (int j = 0; j < 3; j++) {
            for (int k = 0; k < 3; k++) {
                int x = pa[i], y = pb[j], z = pc[k];
                if (x != y && y != z && x != z) {
                    ans = std::max(ans, a[x] + b[y] + c[z]);
                }
            }
        }
    }
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
