#include <bits/stdc++.h>
 using i64 = long long;
using u64 = unsigned long long;
using u32 = unsigned;
using u128 = unsigned __int128;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> a(n);
    int cnt[30] {};
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
        for (int j = 0; j < 30; j++) {
            cnt[j] += a[i] >> j & 1;
        }
    }
        i64 ans = 0;
    for (int i = 0; i < n; i++) {
        i64 res = 0;
        for (int j = 0; j < 30; j++) {
            res += (a[i] >> j & 1 ? n - cnt[j] : cnt[j]) * (1LL << j);
        }
        ans = std::max(ans, res);
    }
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
