#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n, V;
    std::cin >> n >> V;
        std::vector<int> a(n), b(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
    for (int i = 0; i < n; i++) {
        std::cin >> b[i];
    }
        int suma = std::accumulate(a.begin(), a.end(), 0);
    double ans = V;
    for (int i = 0; i < n; i++) {
        ans = std::min(ans, 1. * b[i] / a[i] * suma);
    }
    std::cout << std::fixed << std::setprecision(10) << ans << "\n";
        return 0;
}
