#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int m, s;
    std::cin >> m >> s;
        int max = 0;
    for (int i = 0; i < m; i++) {
        int x;
        std::cin >> x;
        max = std::max(max, x);
        s += x;
    }
        for (int i = max; ; i++) {
        int x = i * (i + 1) / 2;
        if (s < x) {
            std::cout << "NO\n";
            return;
        }
        if (s == x) {
            std::cout << "YES\n";
            return;
        }
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
