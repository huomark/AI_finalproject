#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        int cur = 0;
    int a = 0, b = 0;
    for (int i = 1; cur < n; i++) {
        int t = std::min(i, n - cur);
        if (i / 2 % 2 == 0) {
            a += t;
        } else {
            b += t;
        }
        cur += t;
    }
        std::cout << a << " " << b << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
