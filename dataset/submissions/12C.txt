#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n, m;
    std::cin >> n >> m;
        std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
        std::map<std::string, int> cnt;
    for (int i = 0; i < m; i++) {
        std::string s;
        std::cin >> s;
        cnt[s] += 1;
    }
        std::vector<int> c;
    for (auto [_, x] : cnt) {
        c.push_back(x);
    }
    c.resize(n);
    std::sort(a.begin(), a.end());
    std::sort(c.begin(), c.end());
        int min = 0, max = 0;
    for (int i = 0; i < n; i++) {
        max += a[i] * c[i];
        min += a[i] * c[n - 1 - i];
    }
    std::cout << min << " " << max << "\n";
        return 0;
}
