#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::string s;
    std::cin >> s;
        std::array<int, 26> cnt{};
    for (auto c : s) {
        cnt[c - 'A'] += 1;
    }
    int ans = 0;
    for (int i = 0; i < 26; i++) {
        ans += (cnt[i] > i);
    }
    std::cout << ans << "\n";
}

}
