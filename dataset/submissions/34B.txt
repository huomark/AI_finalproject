#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n, m;
    std::cin >> n >> m;
        std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
    std::sort(a.begin(), a.end());
        int ans = 0;
    for (int i = 0; i < m; i++) {
        ans += std::max(0, -a[i]);
    }
    std::cout << ans << "\n";
        return 0;
}
