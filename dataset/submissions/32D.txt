#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n, m, k;
    std::cin >> n >> m >> k;
        int lim = std::min(n, m) / 2;
    std::vector<std::string> s(n);
    for (int i = 0; i < n; i++) {
        std::cin >> s[i];
    }
    for (int r = 1; r <= lim; r++) {
        for (int i = r; i + r < n; i++) {
            for (int j = r; j + r < m; j++) {
                if (s[i][j] == '*' && s[i - r][j] == '*' && s[i + r][j] == '*' && s[i][j - r] == '*' && s[i][j + r] == '*') {
                    k--;
                    if (k == 0) {
                        std::cout << i + 1 << " " << j + 1 << "\n";
                        std::cout << i - r + 1 << " " << j + 1 << "\n";
                        std::cout << i + r + 1 << " " << j + 1 << "\n";
                        std::cout << i + 1 << " " << j - r + 1 << "\n";
                        std::cout << i + 1 << " " << j + r + 1 << "\n";
                        return 0;
                    }
                }
            }
        }
    }
    std::cout << -1 << "\n";
        return 0;
}
