#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n, t;
    std::cin >> n >> t;
        std::vector<std::pair<int, int>> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i].first >> a[i].second;
    }
        std::sort(a.begin(), a.end());
        int ans = 2;
    for (int i = 1; i < n; i++) {
        int d = 2 * (a[i].first - a[i - 1].first) - a[i].second - a[i - 1].second;
        if (d == 2 * t) {
            ans += 1;
        } else if (d > 2 * t) {
            ans += 2;
        }
    }
        std::cout << ans << "\n";
        return 0;
}
