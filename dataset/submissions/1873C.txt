#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int ans = 0;
    for (int i = 0; i < 10; i++) {
        for (int j = 0; j < 10; j++) {
            char c;
            std::cin >> c;
            if (c == 'X') {
                ans += std::min({i + 1, 10 - i, j + 1, 10 - j});
            }
        }
    }
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
