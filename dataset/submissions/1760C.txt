#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> a(n), x(n + 1), y(n + 1);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
        x[i + 1] = y[i] = a[i];
    }
        for (int i = 1; i <= n; i++) {
        x[i] = std::max(x[i], x[i - 1]);
    }
    for (int i = n - 1; i >= 0; i--) {
        y[i] = std::max(y[i], y[i + 1]);
    }
        for (int i = 0; i < n; i++) {
        std::cout << a[i] - std::max(x[i], y[i + 1]) << " \n"[i == n - 1];
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
