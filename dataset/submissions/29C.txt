#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n;
    std::cin >> n;
        std::map<int, std::vector<int>> adj;
    for (int i = 0; i < n; i++) {
        int a, b;
        std::cin >> a >> b;
        adj[a].push_back(b);
        adj[b].push_back(a);
    }
        int x;
    for (auto [y, e] : adj) {
        if (e.size() == 1) {
            x = y;
        }
    }
        std::cout << x;
    int y = adj[x][0];
    while (true) {
        std::cout << " " << y;
        if (adj[y].size() == 1) {
            break;
        }
        x = adj[y][0] ^ adj[y][1] ^ x;
        std::swap(x, y);
    }
    std::cout << "\n";
        return 0;
}
