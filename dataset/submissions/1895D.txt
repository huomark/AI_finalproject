#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n;
    std::cin >> n;
        std::vector<int> a(n);
    for (int i = 0; i < n - 1; i++) {
        int x;
        std::cin >> x;
        a[i + 1] = a[i] ^ x;
    }
        int k = __builtin_ctz(n);
    std::vector<int> cnt(2 * n + 1), f(2 * n + 1);
    for (int i = 0; i < n; i++) {
        cnt[a[i] >> (k + 1)] += 1;
        f[a[i] >> (k + 1)] = a[i];
    }
    int x = -1;
    for (int i = 0; ; i++) {
        if (cnt[i] == (1 << k)) {
            x = f[i] ^ (n - 1);
            break;
        }
    }
    for (int i = 0; i < n; i++) {
        a[i] ^= x;
        std::cout << a[i] << " \n"[i == n - 1];
    }
        return 0;
}
