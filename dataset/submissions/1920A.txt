#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        int l = 1, r = 1E9;
    std::set<int> s;
    while (n--) {
        int a, x;
        std::cin >> a >> x;
                if (a == 1) {
            l = std::max(l, x);
        } else if (a == 2) {
            r = std::min(r, x);
        } else {
            s.insert(x);
        }
    }
        int ans = std::max(0, r - l + 1);
    for (auto x : s) {
        ans -= (l <= x && x <= r);
    }
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
