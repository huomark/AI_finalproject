#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> p(n);
    for (int i = 0; i < n; i++) {
        std::cin >> p[i];
    }
        int x = std::find(p.begin(), p.end(), 1) - p.begin() + 1;
    int y = std::find(p.begin(), p.end(), 2) - p.begin() + 1;
    int z = std::find(p.begin(), p.end(), n) - p.begin() + 1;
        if (z < std::min(x, y)) {
        std::cout << z << " " << std::min(x, y) << "\n";
    } else if (z > std::max(x, y)) {
        std::cout << z << " " << std::max(x, y) << "\n";
    } else {
        std::cout << 1 << " " << 1 << "\n";
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
     return 0;
}
