#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    std::string s;
    std::cin >> s;
        int n = s.size();
        int carry = 0;
    int k = n;
    for (int i = n - 1; i >= 0; i--) {
        if (s[i] + carry >= '5') {
            k = i;
            carry = 1;
        } else {
            s[i] += carry;
            carry = 0;
        }
    }
    for (int i = k; i < n; i++) {
        s[i] = '0';
    }
    if (carry) {
        s = '1' + s;
    }
    std::cout << s << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
