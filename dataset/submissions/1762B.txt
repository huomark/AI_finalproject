#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> a(n);
    std::cout << n << "\n";
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
        int p = 1 << (std::__lg(a[i]) + 1);
        std::cout << i + 1 << " " << p - a[i] << "\n";
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
