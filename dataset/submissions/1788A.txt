#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> a(n);
    int sum = 0;
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
        sum += a[i] - 1;
    }
        if (sum % 2 == 1) {
        std::cout << -1 << "\n";
        return;
    }
        int s = 0;
    for (int i = 1; i < n; i++) {
        s += a[i - 1] - 1;
        if (s == sum / 2) {
            std::cout << i << "\n";
            return;
        }
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
