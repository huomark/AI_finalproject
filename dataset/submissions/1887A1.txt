#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n, m;
    std::cin >> n >> m;
        std::vector<int> a(n), b(n);
    for (int i = 1; i < n; i++) {
        std::cin >> a[i];
    }
    for (int i = 0; i < n; i++) {
        std::cin >> b[i];
    }
    a[0] = 1;
    std::sort(a.begin(), a.end());
    std::sort(b.begin(), b.end());
        int ans = 0;
    for (int i = 0, j = 0; i < n; i++) {
        while (j < n && a[i] >= b[j]) {
            j++;
        }
        ans = std::max(ans, j - i);
    }
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
