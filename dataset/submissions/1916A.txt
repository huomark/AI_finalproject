#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n, k;
    std::cin >> n >> k;
        int res = 2023;
    std::vector<int> b(n);
    for (int i = 0; i < n; i++) {
        std::cin >> b[i];
    }
        for (int i = 0; i < n; i++) {
        if (res % b[i]) {
            std::cout << "NO\n";
            return;
        }
        res /= b[i];
    }
    std::cout << "YES\n";
    for (int i = 0; i < k; i++) {
        std::cout << (i == 0 ? res : 1) << " \n"[i == k - 1];
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
