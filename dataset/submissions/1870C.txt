#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n, k;
    std::cin >> n >> k;
        std::vector<int> a(n);
    std::vector<int> l(k, n), r(k, -1);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
        a[i]--;
        l[a[i]] = std::min(l[a[i]], i);
        r[a[i]] = i;
    }
    auto sl = l, sr = r;
    for (int i = k - 2; i >= 0; i--) {
        sl[i] = std::min(sl[i], sl[i + 1]);
        sr[i] = std::max(sr[i], sr[i + 1]);
    }
    for (int i = 0; i < k; i++) {
        if (l[i] > r[i]) {
            std::cout << 0;
        } else {
            std::cout << 2 * (sr[i] - sl[i] + 1);
        }
        std::cout << " \n"[i == k - 1];
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
