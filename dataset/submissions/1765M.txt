#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        int ans = 1;
    for (int d = 2; d * d <= n; d++) {
        if (n % d == 0) {
            ans = std::max(ans, n / d);
        }
    }
        std::cout << ans << " " << n - ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
