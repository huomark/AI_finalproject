#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n, h;
    std::cin >> n >> h;
        std::vector<int> x(n);
    for (int i = 0; i < n; i++) {
        std::cin >> x[i];
    }
        constexpr i64 inf = 1E18;
        std::vector dp(n, std::vector<i64>(n, inf));
        for (int i = 0; i < n; i++) {
        dp[i][i] = h;
    }
    for (int l = n - 1; l >= 0; l--) {
        for (int r = l + 1; r < n; r++) {
            for (int m = l; m < r; m++) {
                dp[l][r] = std::min(dp[l][r], dp[l][m] + dp[m + 1][r] - std::max(0, h + 1 - (x[r] - x[l] + 1) / 2));
            }
        }
    }
    std::cout << dp[0][n - 1] << "\n";
        return 0;
}
