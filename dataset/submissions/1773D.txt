#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
        int min = *std::min_element(a.begin(), a.end());
    int max = *std::max_element(a.begin(), a.end());
    int cmin = std::count(a.begin(), a.end(), min);
    int cmax = std::count(a.begin(), a.end(), max);
        if (min == max) {
        std::cout << 1LL * n * (n - 1) << "\n";
    } else {
        std::cout << 2LL * cmin * cmax << "\n";
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
