#include <bits/stdc++.h>
 using i64 = long long;
using u64 = unsigned long long;
using u32 = unsigned;
using u128 = unsigned __int128;
 void solve() {
    int n, m;
    std::cin >> n >> m;
        std::vector<i64> a(n);
    i64 ans = 0;
    for (int i = 0; i < n; i++) {
        i64 sum = 0;
        for (int j = 0; j < m; j++) {
            int x;
            std::cin >> x;
            ans += 1LL * x * (m - j);
            sum += x;
        }
        a[i] = sum;
    }
    std::sort(a.begin(), a.end());
    for (int i = 0; i < n; i++) {
        ans += a[i] * i * m;
    }
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
