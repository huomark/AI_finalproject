#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n, m, x1, y1, x2, y2;
    std::cin >> n >> m >> x1 >> y1 >> x2 >> y2;
        int dx = std::abs(x1 - x2);
    int dy = std::abs(y1 - y2);
        i64 ans = 2LL * (n - dx) * (m - dy);
    ans -= 1LL * std::max(0, n - 2 * dx) * std::max(0, m - 2 * dy);
    ans = 1LL * n * m - ans;
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
