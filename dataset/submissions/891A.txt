#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n;
    std::cin >> n;
        int ans = n + 1;
    std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
    for (int i = 0; i < n; i++) {
        int g = 0;
        for (int j = i; j < n; j++) {
            g = std::gcd(g, a[j]);
            if (g == 1) {
                ans = std::min(ans, j - i);
                break;
            }
        }
    }
    if (ans > n) {
        std::cout << -1 << "\n";
        return 0;
    }
        int cnt = std::count(a.begin(), a.end(), 1);
    if (cnt > 0) {
        ans = n - cnt;
    } else {
        ans += n - 1;
    }
    std::cout << ans << "\n";
        return 0;
}
