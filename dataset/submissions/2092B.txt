#include <bits/stdc++.h>
 using i64 = long long;
using u64 = unsigned long long;
using u32 = unsigned;
 void solve() {
    int n;
    std::cin >> n;
        std::string a, b;
    std::cin >> a >> b;
        int cnt[2] {};
    for (int i = 0; i < n; i++) {
        if (a[i] == '0') {
            cnt[i % 2]++;
        }
        if (b[i] == '0') {
            cnt[(i + 1) % 2]++;
        }
    }
        if (cnt[0] >= (n + 1) / 2 && cnt[1] >= n / 2) {
        std::cout << "YES\n";
    } else {
        std::cout << "NO\n";
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
