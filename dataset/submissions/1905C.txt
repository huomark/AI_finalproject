#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::string s;
    std::cin >> s;
        std::vector<int> a;
    for (int i = n - 1; i >= 0; i--) {
        if (a.empty() || s[i] >= s[a.back()]) {
            a.push_back(i);
        }
    }
    std::reverse(a.begin(), a.end());
    int j = 0;
    while (j < a.size() && s[a[0]] == s[a[j]]) {
        j += 1;
    }
    int ans = a.size() - j;
    for (int i = 0; i < a.size() - 1 - i; i++) {
        std::swap(s[a[i]], s[a[a.size() - 1 - i]]);
    }
    if (!std::is_sorted(s.begin(), s.end())) {
        ans = -1;
    }
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
