#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n;
    std::cin >> n;
        std::vector<double> p(n);
    for (int i = 0; i < n; i++) {
        std::cin >> p[i];
    }
        double ans = 0;
    std::cout << std::fixed << std::setprecision(10);
        double s = 0;
    for (int i = 0; i < n; i++) {
        s = (s + 1) * p[i];
        ans += s * 2;
        ans -= p[i];
    }
    std::cout << ans << "\n";
        return 0;
}
