#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
        i64 seg = 0;
    i64 sum = 0;
    i64 ans = 0;
    i64 max = 0;
        for (int i = 0; i < n; i++) {
        sum += a[i];
        max = std::max(max, sum);
        if (sum - max < seg) {
            seg = sum - max;
            ans = max;
        }
    }
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
