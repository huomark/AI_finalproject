#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
        int q;
    std::cin >> q;
        std::vector<int> ans(q);
    std::vector<std::array<int, 3>> qry(q);
    for (int i = 0; i < q; i++) {
        int x, k;
        std::cin >> x >> k;
        x--;
        k = std::min(k, 20);
        qry[i] = {k, x, i};
    }
    std::sort(qry.begin(), qry.end());
        int cur = 0;
    for (auto [k, x, i] : qry) {
        while (cur < k) {
            std::vector<int> cnt(n + 1);
            for (auto x : a) {
                cnt[x]++;
            }
            for (auto &x : a) {
                x = cnt[x];
            }
            cur++;
        }
        ans[i] = a[x];
    }
    for (int i = 0; i < q; i++) {
        std::cout << ans[i] << "\n";
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
