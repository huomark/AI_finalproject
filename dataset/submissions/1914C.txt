#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n, k;
    std::cin >> n >> k;
        std::vector<int> a(n), b(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
    for (int i = 0; i < n; i++) {
        std::cin >> b[i];
    }
        i64 ans = 0;
    i64 sum = 0;
    int max = 0;
    for (int i = 0; i < n && i + 1 <= k; i++) {
        sum += a[i];
        max = std::max(max, b[i]);
        ans = std::max(ans, sum + 1LL * max * (k - i - 1));
    }
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
