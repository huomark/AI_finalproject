#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        std::string s, t;
        std::getline(std::cin, s);
    std::getline(std::cin, t);
        std::map<char, int> c;
    for (auto x : s) {
        if (x != ' ') {
            c[x]++;
        }
    }
    for (auto x : t) {
        if (x != ' ') {
            c[x]--;
        }
    }
        for (auto [_, x] : c) {
        if (x < 0) {
            std::cout << "NO\n";
            return 0;
        }
    }
    std::cout << "YES\n";
        return 0;
}
