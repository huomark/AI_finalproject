#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n, d;
    std::cin >> n >> d;
        std::string s;
    std::cin >> s;
        int i = 0;
    while (i < n && s[i] >= '0' + d) {
        i += 1;
    }
    s.insert(s.begin() + i, '0' + d);
    std::cout << s << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
