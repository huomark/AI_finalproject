#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n;
    std::cin >> n;
        std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
        int ans = 2E9;
    auto pre = a, suf = a;
    for (int i = 0; i < n; i++) {
        pre[i] -= i;
        if (i) {
            pre[i] = std::max(pre[i], pre[i - 1]);
        }
    }
    for (int i = n - 1; i >= 0; i--) {
        suf[i] += i;
        if (i < n - 1) {
            suf[i] = std::max(suf[i], suf[i + 1]);
        }
    }
    for (int i = 0; i < n; i++) {
        int res = a[i];
        if (i > 0) {
            res = std::max(res, pre[i - 1] + n - 1);
        }
        if (i < n - 1) {
            res = std::max(res, suf[i + 1]);
        }
        ans = std::min(ans, res);
    }
    std::cout << ans << "\n";
        return 0;
}
