#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<std::string> s(n);
    for (int i = 0; i < n; i++) {
        std::cin >> s[i];
    }
        for (int i = 0; i < n; i++) {
        if (std::count(s[i].begin(), s[i].end(), '1') == 1) {
            std::cout << "TRIANGLE\n";
            return;
        }
    }
    std::cout << "SQUARE\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
