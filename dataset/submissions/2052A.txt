#include <bits/stdc++.h>
 using i64 = long long;
using u64 = unsigned long long;
using u32 = unsigned;
using u128 = unsigned __int128;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n;
    std::cin >> n;
        std::vector<int> c(n);
    for (int i = 0; i < n; i++) {
        std::cin >> c[i];
        c[i]--;
    }
        std::vector<std::array<int, 2>> ans;
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n - i - 1; j++) {
            if (c[j] < c[j + 1]) {
                ans.push_back({c[j + 1], c[j]});
                std::swap(c[j + 1], c[j]);
            }
        }
    }
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n - i - 1; j++) {
            if (c[j] > c[j + 1]) {
                ans.push_back({c[j + 1], c[j]});
                std::swap(c[j + 1], c[j]);
            }
        }
    }
    std::reverse(ans.begin(), ans.end());
        std::cout << ans.size() << "\n";
    for (auto [x, y] : ans) {
        std::cout << y + 1 << " " << x + 1 << "\n";
    }
        return 0;
}
