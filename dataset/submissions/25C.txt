#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n;
    std::cin >> n;
        std::vector d(n, std::vector<int>(n));
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            std::cin >> d[i][j];
        }
    }
        int k;
    std::cin >> k;
        while (k--) {
        int a, b, c;
        std::cin >> a >> b >> c;
        a--, b--;
                i64 ans = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                d[i][j] = std::min({d[i][j], d[i][a] + c + d[b][j], d[i][b] + c + d[a][j]});
                if (i < j) {
                    ans += d[i][j];
                }
            }
        }
        std::cout << ans << "\n";
    }
        return 0;
}
