#include <bits/stdc++.h>
 using i64 = long long;
 bool isVowel(char c) {
    return c == 'a' || c == 'e';
}
 void solve() {
    int n;
    std::cin >> n;
        std::string s;
    std::cin >> s;
        std::vector<std::string> t;
    for (int i = n - 1; i >= 0; ) {
        if (isVowel(s[i])) {
            t.push_back(s.substr(i - 1, 2));
            i -= 2;
        } else {
            t.push_back(s.substr(i - 2, 3));
            i -= 3;
        }
    }
    std::reverse(t.begin(), t.end());
    for (int i = 0; i < t.size(); i++) {
        std::cout << t[i] << ".\n"[i == t.size() - 1];
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
