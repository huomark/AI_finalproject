#include <bits/stdc++.h>
 void solve() {
    int n;
    std::cin >> n;
        if (n == 2) {
        std::cout << "YES\n";
        std::cout << 1 << " " << 1 << "\n";
        return;
    }
        if (n == 3) {
        std::cout << "NO\n";
        return;
    }
        std::cout << "YES\n";
    for (int i = 0; i < n; i++) {
        std::cout << (i & 1 ? (n - 1) / 2 : -(n - 2) / 2) << " \n"[i == n - 1];
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
