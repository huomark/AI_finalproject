#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n, t;
    std::cin >> n >> t;
        std::vector<int> b(n);
    int ans = 0;
    for (int i = 0; i < n; i++) {
        std::cin >> b[i];
        if (i > 0 && b[i] <= b[i - 1]) {
            int v = (b[i - 1] - b[i]) / t + 1;
            ans += v;
            b[i] += v * t;
        }
    }
    std::cout << ans << "\n";
        return 0;
}
