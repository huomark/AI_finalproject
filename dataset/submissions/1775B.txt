#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::map<int, int> cnt;
        std::vector<std::vector<int>> c(n);
    for (int i = 0; i < n; i++) {
        int k;
        std::cin >> k;
                c[i].resize(k);
        for (auto &x : c[i]) {
            std::cin >> x;
            cnt[x]++;
        }
    }
        for (int i = 0; i < n; i++) {
        int ok = 1;
        for (auto x : c[i]) {
            if (cnt[x] == 1) {
                ok = 0;
                break;
            }
        }
        if (ok) {
            std::cout << "Yes\n";
            return;
        }
    }
        std::cout << "No\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
