#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        if (n == 1) {
        std::cout << 1 << "\n";
        return;
    }
        if (n == 2) {
        std::cout << 1 << " " << 2 << "\n";
        return;
    }
        std::vector<int> p(n);
    p[0] = 2, p[n - 1] = 3;
    p[n / 2] = 1;
    int x = 3;
    for (int i = 0; i < n; i++) {
        if (p[i] == 0) {
            p[i] = ++x;
        }
        std::cout << p[i] << " \n"[i == n - 1];
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
