#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
        std::vector<int> l(n), r(n);
    for (int i = 1; i < n; i++) {
        if (i == n - 1 || a[i + 1] - a[i] > a[i] - a[i - 1]) {
            l[i] = l[i - 1] + 1;
        } else {
            l[i] = l[i - 1] + a[i] - a[i - 1];
        }
    }
    r[n - 1] = 0;
    for (int i = n - 2; i >= 0; i--) {
        if (i == 0 || a[i] - a[i - 1] > a[i + 1] - a[i]) {
            r[i] = r[i + 1] + 1;
        } else {
            r[i] = r[i + 1] + a[i + 1] - a[i];
        }
    }
        int q;
    std::cin >> q;
        while (q--) {
        int x, y;
        std::cin >> x >> y;
        x--, y--;
                int ans;
        if (x < y) {
            ans = r[x] - r[y];
        } else {
            ans = l[x] - l[y];
        }
        std::cout << ans << "\n";
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
