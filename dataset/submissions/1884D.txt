#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        i64 ans = 0;
    std::vector<i64> f(n + 1);
    std::vector<int> g(n + 1);
    for (int i = 0; i < n; i++) {
        int a;
        std::cin >> a;
        f[a] += 1;
        g[a] = 1;
    }
    for (int i = 1; i <= n; i++) {
        for (int j = 2 * i; j <= n; j += i) {
            g[j] |= g[i];
            f[i] += f[j];
        }
    }
    for (int i = 1; i <= n; i++) {
        f[i] *= f[i];
    }
    for (int i = n; i; i--) {
        for (int j = 2 * i; j <= n; j += i) {
            f[i] -= f[j];
        }
    }
    for (int i = 1; i <= n; i++) {
        if (!g[i]) {
            ans += f[i];
        }
    }
    ans /= 2;
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
