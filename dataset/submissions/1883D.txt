#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int q;
    std::cin >> q;
        std::multiset<int> ls, rs;
        while (q--) {
        char op;
        int l, r;
        std::cin >> op >> l >> r;
                if (op == '+') {
            ls.insert(l);
            rs.insert(r);
        } else {
            ls.extract(l);
            rs.extract(r);
        }
                if (!ls.empty() && *ls.rbegin() > *rs.begin()) {
            std::cout << "YES\n";
        } else {
            std::cout << "NO\n";
        }
    }
        return 0;
}
