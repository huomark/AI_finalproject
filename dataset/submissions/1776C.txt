#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n, m;
    std::cin >> n >> m;
        std::vector<int> y(n);
    for (int i = 0; i < n; i++) {
        std::cin >> y[i];
    }
    std::sort(y.begin(), y.end(), std::greater());
        int t = -1;
    for (int i = 0; i < n; i++) {
        if (y[i] * (i + 1) > m) {
            t = i;
            break;
        }
    }
        std::vector<int> a{0, m + 1};
        if (t == -1) {
        std::cout << "Alessia" << std::endl;
        for (int i = 0; i < n; i++) {
            int len = y[i];
            for (int j = 0; j < a.size() - 1; j++) {
                if (a[j + 1] - a[j] > len) {
                    std::cout << len << " " << a[j] + 1 << std::endl;
                    break;
                }
            }
            int b;
            std::cin >> b;
            a.push_back(b);
            std::sort(a.begin(), a.end());
        }
    } else {
        std::cout << "Bernardo" << std::endl;
        for (int i = 0; i < n; i++) {
            int len, l;
            std::cin >> len >> l;
                        int r = l + len - 1;
                        bool ok = false;
            for (auto x : a) {
                if (l <= x && x <= r) {
                    std::cout << x << std::endl;
                    ok = true;
                    break;
                }
            }
                        if (!ok) {
                for (int i = y[t]; i <= m; i += y[t]) {
                    if (l <= i && i <= r) {
                        std::cout << i << std::endl;
                        ok = true;
                        a.push_back(i);
                        break;
                    }
                }
                if (!ok) {
                    std::cout << l << std::endl;
                }
            }
        }
    }
        return 0;
}
