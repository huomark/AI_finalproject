#include <bits/stdc++.h>
 using i64 = long long;
using u64 = unsigned long long;
using u32 = unsigned;
using u128 = unsigned __int128;
 void solve() {
    int w, b;
    std::cin >> w >> b;
        int k = std::sqrt(2LL * (w + b));
    if (1LL * k * (k + 1) / 2 > w + b) {
        k--;
    }
    std::cout << k << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
