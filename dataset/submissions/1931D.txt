#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n, x, y;
    std::cin >> n >> x >> y;
        std::map<std::pair<int, int>, int> cnt;
    i64 ans = 0;
    for (int i = 0; i < n; i++) {
        int a;
        std::cin >> a;
                ans += cnt[{(x - a % x) % x, a % y}];
        cnt[{a % x, a % y}] += 1;
    }
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
