#include <bits/stdc++.h>
 using i64 = long long;
using u64 = unsigned long long;
using u32 = unsigned;
using u128 = unsigned __int128;
 void solve() {
    int n, x, y;
    std::cin >> n >> x >> y;
    x--;
    y--;
        std::vector<int> a(n);
    a[x] = 0;
    a[y] = 1;
    for (int i = x; i != y; i = (i + 1) % n) {
        a[i] = (i - x + n) % n % 2;
        if ((i + 1) % n == y && a[i] == 1) {
            a[i] = 2;
        }
    }
    for (int i = x; i != y; i = (i - 1 + n) % n) {
        a[i] = (x - i + n) % n % 2;
        if ((i - 1 + n) % n == y && a[i] == 1) {
            a[i] = 2;
        }
    }
    for (int i = 0; i < n; i++) {
        std::cout << a[i] << " \n"[i == n - 1];
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
