#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<std::vector<int>> a(n);
    int min = 1E9;
    int min2 = 1E9;
    i64 ans = 0;
    for (int i = 0; i < n; i++) {
        int m;
        std::cin >> m;
        a[i].resize(m);
        for (int j = 0; j < m; j++) {
            std::cin >> a[i][j];
        }
        std::nth_element(a[i].begin(), a[i].begin() + 1, a[i].end());
        min = std::min(min, a[i][0]);
        min2 = std::min(min2, a[i][1]);
        ans += a[i][1];
    }
    ans -= min2;
    ans += min;
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
