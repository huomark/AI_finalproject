#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int f[3][3] {};
    for (int i = 0; i < 3; i++) {
        std::string s;
        std::cin >> s;
                if (s[1] == '<') {
            f[s[0] - 'A'][s[2] - 'A'] = 1;
        } else {
            f[s[2] - 'A'][s[0] - 'A'] = 1;
        }
    }
        int p[] = {0, 1, 2};
    do {
        bool ok = true;
        for (int i = 0; i < 3; i++) {
            for (int j = i+1; j < 3; j++) {
                if (f[p[j]][p[i]]) {
                    ok = false;
                }
            }
        }
        if (ok) {
            std::string s;
            s += 'A' + p[0];
            s += 'A' + p[1];
            s += 'A' + p[2];
            std::cout << s << "\n";
            return 0;
        }
    } while (std::next_permutation(p, p + 3));
    std::cout << "Impossible\n";
        return 0;
}
