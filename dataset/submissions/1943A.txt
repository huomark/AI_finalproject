#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> cnt(n + 1);
    for (int i = 0; i < n; i++) {
        int a;
        std::cin >> a;
        cnt[a]++;
    }
        int ans = n;
    for (int x = 1; x <= 2; x++) {
        int k = x;
        for (int i = 0; i <= n; i++) {
            if (cnt[i] < x) {
                k--;
                if (k == 0) {
                    ans = std::min(ans, i);
                    break;
                }
            }
        }
    }
        std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
