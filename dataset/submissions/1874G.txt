#include <bits/stdc++.h>
 using i64 = long long;
 constexpr int V = 200;
constexpr int inf = 1E9;
 template<class T>
void chmax(T &a, T b) {
    if (a < b) {
        a = b;
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n, m;
    std::cin >> n >> m;
        std::vector<std::array<int, 3>> e(n);
    for (int i = 0; i < n; i++) {
        int o;
        std::cin >> o;
        e[i][0] = o;
                if (o == 1) {
            std::cin >> e[i][1] >> e[i][2];
        } else if (o > 1) {
            std::cin >> e[i][1];
        }
    }
        std::vector<std::vector<int>> adj(n);
    for (int i = 0; i < m; i++) {
        int u, v;
        std::cin >> u >> v;
        u--, v--;
        adj[u].push_back(v);
    }
        std::vector f(n, std::vector(V + 1, std::vector<int>(V + 1, -inf)));
    std::vector g1(n, std::vector(V + 1, std::vector<int>(V + 1, -inf)));
    std::vector g2(n, std::vector(V + 1, std::vector<int>(V + 1, -inf)));
    std::vector h1(n, std::vector(V + 1, std::vector<int>(V + 1, -inf)));
    std::vector h2(n, std::vector(V + 1, std::vector<int>(V + 1, -inf)));
    std::vector<int> dp(n);
        for (int i = 0; i < n; i++) {
        for (int a = V; a >= 0; a--) {
            for (int b = V; b >= 0; b--) {
                chmax(dp[i], f[i][a][b]);
            }
        }
        for (int a = 0; a <= V; a++) {
            chmax(dp[i], g1[i][a][a]);
            chmax(dp[i], g2[i][a][a]);
            chmax(f[i][a][0], g1[i][a][a]);
            chmax(f[i][0][a], g2[i][a][a]);
            chmax(dp[i], h1[i][0][a]);
            chmax(dp[i], h2[i][0][a]);
            chmax(f[i][a][0], h1[i][0][a]);
            chmax(f[i][0][a], h2[i][0][a]);
            for (int b = a; b <= V; b++) {
                chmax(h1[i][b - a][0], g1[i][a][b]);
                chmax(h2[i][b - a][0], g2[i][a][b]);
            }
        }
        chmax(f[i][0][0], dp[i]);
        for (int a = V; a >= 0; a--) {
            for (int b = V; b >= 0; b--) {
                if (e[i][0] == 1) {
                    chmax(f[i][std::max(a, e[i][1])][std::max(b, e[i][2])], f[i][a][b] + e[i][1] * e[i][2]);
                } else if (e[i][0] == 2) {
                    f[i][a][b] += b * e[i][1];
                } else if (e[i][0] == 3) {
                    f[i][a][b] += a * e[i][1];
                } else if (e[i][0] == 4) {
                    f[i][a][b] += e[i][1];
                }
            }
        }
        if (e[i][0] == 1) {
            for (int a = V; a >= 0; a--) {
                for (int b = V; b >= a; b--) {
                    g1[i][a][b] += e[i][1] * e[i][2];
                    g2[i][a][b] += e[i][1] * e[i][2];
                }
            }
            for (int b = e[i][1]; b <= V; b++) {
                chmax(g1[i][e[i][1]][b], dp[i] + e[i][2] * b);
            }
            for (int b = e[i][2]; b <= V; b++) {
                chmax(g2[i][e[i][2]][b], dp[i] + e[i][1] * b);
            }
            for (int a = 0; a <= V; a++) {
                for (int b = V; b >= 0; b--) {
                    chmax(h1[i][a][std::max(b, e[i][1])], h1[i][a][b] + e[i][1] * e[i][2]);
                    chmax(h2[i][a][std::max(b, e[i][2])], h2[i][a][b] + e[i][1] * e[i][2]);
                }
            }
        } else if (e[i][0] == 2) {
            for (int a = V; a >= 0; a--) {
                for (int b = V; b >= a; b--) {
                    if (a + e[i][1] <= b) {
                        chmax(g1[i][a + e[i][1]][b], g1[i][a][b]);
                    }
                    g2[i][a][b] += e[i][1] * b;
                }
            }
            for (int a = 0; a <= V; a++) {
                for (int b = V; b >= 0; b--) {
                    if (a >= e[i][1]) {
                        chmax(h1[i][a - e[i][1]][b], h1[i][a][b]);
                    }
                    h2[i][a][b] += e[i][1] * b;
                }
            }
        } else if (e[i][0] == 3) {
            for (int a = V; a >= 0; a--) {
                for (int b = V; b >= a; b--) {
                    if (a + e[i][1] <= b) {
                        chmax(g2[i][a + e[i][1]][b], g2[i][a][b]);
                    }
                    g1[i][a][b] += e[i][1] * b;
                }
            }
            for (int a = 0; a <= V; a++) {
                for (int b = V; b >= 0; b--) {
                    if (a >= e[i][1]) {
                        chmax(h2[i][a - e[i][1]][b], h2[i][a][b]);
                    }
                    h1[i][a][b] += e[i][1] * b;
                }
            }
        } else if (e[i][0] == 4) {
            for (int a = V; a >= 0; a--) {
                for (int b = V; b >= a; b--) {
                    g1[i][a][b] += e[i][1];
                    g2[i][a][b] += e[i][1];
                }
            }
            for (int a = 0; a <= V; a++) {
                for (int b = V; b >= 0; b--) {
                    h1[i][a][b] += e[i][1];
                    h2[i][a][b] += e[i][1];
                }
            }
        }
        for (auto j : adj[i]) {
            for (int a = V; a >= 0; a--) {
                for (int b = V; b >= 0; b--) {
                    chmax(f[j][a][b], f[i][a][b]);
                    chmax(h1[j][a][b], h1[i][a][b]);
                    chmax(h2[j][a][b], h2[i][a][b]);
                }
            }
            for (int a = V; a >= 0; a--) {
                for (int b = V; b >= a; b--) {
                    chmax(g1[j][a][b], g1[i][a][b]);
                    chmax(g2[j][a][b], g2[i][a][b]);
                }
            }
        }
    }
        std::vector suf(n, std::vector<std::pair<int, int>>(n * V + 1, {-inf, -inf}));
    suf[n - 1][0] = {0, 0};
    for (int i = n - 1; i >= 0; i--) {
        for (auto j : adj[i]) {
            for (int a = 0; a <= n * V; a++) {
                chmax(suf[i][a], suf[j][a]);
            }
        }
        if (e[i][0] == 1) {
            for (int a = 0; a <= n * V; a++) {
                suf[i][a].second += e[i][1] * e[i][2];
            }
        } else if (e[i][0] == 2) {
            for (int a = n * V; a >= 0; a--) {
                if (a + e[i][1] <= n * V) {
                    chmax(suf[i][a + e[i][1]], suf[i][a]);
                }
            }
        } else if (e[i][0] == 3) {
            for (int a = 0; a <= n * V; a++) {
                suf[i][a].first += e[i][1];
            }
        } else if (e[i][0] == 4) {
            for (int a = 0; a <= n * V; a++) {
                suf[i][a].second += e[i][1];
            }
        }
    }
        i64 ans = dp[n - 1];
    for (int i = 0; i < n; i++) {
        if (e[i][0] == 1) {
            i64 val = 0;
            for (int a = 0; a <= n * V; a++) {
                if (suf[i][a].first >= 0 && suf[i][a].second >= 0) {
                    int b = suf[i][a].first;
                    int t = suf[i][a].second - e[i][1] * e[i][2];
                    chmax(val, t + 1LL * inf * (e[i][1] + a) * (e[i][2] + b));
                }
            }
            chmax(ans, dp[i] + val);
        }
    }
        std::cout << ans << "\n";
        return 0;
}
