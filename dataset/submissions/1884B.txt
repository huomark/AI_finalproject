#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::string s;
    std::cin >> s;
        i64 ans = 0;
        int l = n, r = n;
    for (int i = 1; i <= n; i++) {
        if (ans != -1) {
            if (l > n - i && s[n - i] == '0') {
                l -= 1;
            } else {
                while (l > 0 && s[l - 1] == '1') {
                    l -= 1;
                }
                if (l == 0) {
                    ans = -1;
                } else {
                    ans += r - l;
                }
                l -= 1;
            }
            r -= 1;
        }
        std::cout << ans << " \n"[i == n];
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
