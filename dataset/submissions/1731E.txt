#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    i64 m;
    std::cin >> n >> m;
        i64 ans = m;
        std::vector<i64> f(n + 1);
    for (int i = 1; i <= n; i++) {
        f[i] = 1LL * (n / i) * (n / i);
    }
    for (int i = n; i; i--) {
        for (int j = 2 * i; j <= n; j += i) {
            f[i] -= f[j];
        }
    }
    for (int i = 1; i <= n; i++) {
        f[i]--;
        f[i] /= 2;
    }
        for (int i = n; i >= 2; i--) {
        i64 t = std::min(f[i], m) / (i - 1);
        m -= t * (i - 1);
        ans += t;
    }
        if (m) ans = -1;
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
