#include <bits/stdc++.h>
 using i64 = long long;
using u64 = unsigned long long;
using u32 = unsigned;
using u128 = unsigned __int128;
 void solve() {
    int x;
    std::cin >> x;
        if ((x & (x - 1)) == 0 || (x & (x + 1)) == 0) {
        std::cout << -1 << "\n";
        return;
    }
        int a = std::__lg(x);
    int b = __builtin_ctz(~x);
        int y = x ^ (1 << a) ^ (1 << b);
    std::cout << y << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
