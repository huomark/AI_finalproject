#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n, k;
    std::cin >> n >> k;
        int p = 1;
    while (true) {
        int t = (n + 1) / 2;
        if (k <= t) {
            std::cout << (2 * k - 1) * p << "\n";
            return;
        }
        k -= t;
        n /= 2;
        p *= 2;
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
