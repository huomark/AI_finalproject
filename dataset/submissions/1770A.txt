#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n, m;
    std::cin >> n >> m;
        std::priority_queue<int> q;
    for (int i = 0; i < n; i++) {
        int x;
        std::cin >> x;
        q.push(-x);
    }
        for (int i = 0; i < m; i++) {
        int x;
        std::cin >> x;
        q.pop();
        q.push(-x);
    }
        i64 ans = 0;
    for (int i = 0; i < n; i++) {
        ans -= q.top();
        q.pop();
    }
        std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
