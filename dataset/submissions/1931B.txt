#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
        i64 sum = std::accumulate(a.begin(), a.end(), 0LL);
        bool ok = true;
    i64 s = 0;
    for (int i = 0; i < n; i++) {
        s += a[i];
        if (s < 1LL * sum / n * (i + 1)) {
            ok = false;
        }
    }
    std::cout << (ok ? "YES" : "NO") << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
