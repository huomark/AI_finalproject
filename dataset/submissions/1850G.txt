#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::map<int, int> f[4];
    i64 ans = 0;
    for (int i = 0; i < n; i++) {
        int x, y;
        std::cin >> x >> y;
        ans += f[0][x]++;
        ans += f[1][y]++;
        ans += f[2][x + y]++;
        ans += f[3][x - y]++;
    }
    std::cout << ans * 2 << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
