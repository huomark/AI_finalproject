#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n, d, h;
    std::cin >> n >> d >> h;
        double ans = 0;
        ans = 1.0 * d * h / 2 * n;
        std::vector<int> y(n);
    for (int i = 0; i < n; i++) {
        std::cin >> y[i];
    }
        for (int i = 1; i < n; i++) {
        int a = h - (y[i] - y[i - 1]);
        if (a > 0) {
            ans -= 1.0 * a * a / h * d / 2;
        }
    }
        std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        std::cout << std::fixed << std::setprecision(10);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
