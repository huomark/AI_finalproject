#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n, m, k;
    std::cin >> n >> m >> k;
        std::vector<int> f(m, 1E9);
    for (int i = 0; i < n; i++) {
        int x, y;
        std::cin >> x >> y;
        x--;
        f[x] = std::min(f[x], y);
    }
    int ans = std::min(std::accumulate(f.begin(), f.end(), 0), k);
    std::cout << ans << "\n";
        return 0;
}
