#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        int cnt[2] {};
    for (int i = 0; i < n; i++) {
        int a;
        std::cin >> a;
        if (a <= 1) {
            cnt[a]++;
        }
    }
        i64 ans = (1LL * cnt[1]) << cnt[0];
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
