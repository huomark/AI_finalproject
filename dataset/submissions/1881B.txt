#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int a, b, c;
    std::cin >> a >> b >> c;
        for (auto x : {a, a / 2, a / 3, a / 4}) {
        if (x == 0) {
            continue;
        }
        bool ok = true;
        int cnt = 0;
        for (auto y : {a, b, c}) {
            if (y % x != 0) {
                ok = false;
                break;
            }
            cnt += y / x - 1;
        }
        if (ok && cnt <= 3) {
            std::cout << "YES\n";
            return;
        }
    }
    std::cout << "NO\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
