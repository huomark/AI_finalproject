#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n, m, v;
    std::cin >> n >> m >> v;
        v--;
        if (m < n - 1 || m > 1 + (n - 1) * (n - 2) / 2) {
        std::cout << -1 << "\n";
        return 0;
    }
        std::vector<std::pair<int, int>> ans;
    for (int i = 0; i < n; i++) {
        if (v != i) {
            ans.emplace_back(v, i);
        }
    }
    int x = v ? 0 : 1;
    for (int i = 0; i < n && ans.size() < m; i++) {
        for (int j = i + 1; j < n && ans.size() < m; j++) {
            if (i != v && j != v && i != x && j != x) {
                ans.emplace_back(i, j);
            }
        }
    }
    for (auto [x, y] : ans) {
        std::cout << x + 1 << " " << y + 1 << "\n";
    }
        return 0;
}
