#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    std::string s;
    std::cin >> s;
        for (int i = 1; i < s.size(); i++) {
        std::string a = s.substr(0, i);
        std::string b = s.substr(i);
        if (b[0] != '0' && std::stoi(b) > std::stoi(a)) {
            std::cout << a << " " << b << "\n";
            return;
        }
    }
    std::cout << -1 << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
