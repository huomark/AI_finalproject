#include <bits/stdc++.h>
 using i64 = long long;
using u64 = unsigned long long;
using u32 = unsigned;
using u128 = unsigned __int128;
 void solve() {
    int n, m, r, c;
    std::cin >> n >> m >> r >> c;
        i64 ans = 1LL * n * m - 1LL * (r - 1) * m - c;
    ans += 1LL * (m - 1) * (n - r);
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    std::cin >> t;
        while (t--) {
        solve();
    }

}
