#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::string s;
    std::cin >> s;
        int i = 0;
    for (int j = 0; j < n; j++) {
        if (s[j] <= s[i]) {
            i = j;
        }
    }
        s = s[i] + s.substr(0, i) + s.substr(i + 1);
    std::cout << s << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
