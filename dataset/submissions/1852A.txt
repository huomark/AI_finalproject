#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n, k;
    std::cin >> n >> k;
        std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
        a[i]--;
    }
        i64 ans = 0;
    for (int i = 0; i < n; i++) {
        if (ans >= a[i]) {
            if (ans >= a[i] + 1LL * i * (k - 1)) {
                ans += k;
            } else {
                ans += (ans - a[i]) / i + 1;
            }
        }
    }
    std::cout << ans + 1 << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
