#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n, k, m;
    std::cin >> n >> k >> m;
        std::string s;
    std::cin >> s;
        std::string t;
    int j = 0;
    bool ok = true;
    for (int i = 0; i < n; i++) {
        int cnt = 0;
        std::vector<int> occ(k);
        while (cnt < k && j < m) {
            cnt += !occ[s[j] - 'a'];
            occ[s[j] - 'a'] = 1;
            j += 1;
        }
        if (cnt < k) {
            ok = false;
            for (int x = 0; x < k; x++) {
                if (!occ[x]) {
                    t += 'a' + x;
                    break;
                }
            }
        } else {
            t += s[j - 1];
        }
    }
        if (ok) {
        std::cout << "YES\n";
    } else {
        std::cout << "NO\n";
        std::cout << t << "\n";
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
