#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> cnt(n + 1);
    for (int i = 0; i < n; i++) {
        int a;
        std::cin >> a;
        cnt[a]++;
    }
    i64 ans = 0;
    int tot = 0;
    for (int i = 0; i <= n; i++) {
        ans += 1LL * cnt[i] * (cnt[i] - 1) * (cnt[i] - 2) / 6;
        ans += 1LL * cnt[i] * (cnt[i] - 1) / 2 * tot;
        tot += cnt[i];
    }
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
