#include <bits/stdc++.h>
 using i64 = long long;
    #ifdef ONLINE_JUDGE
    std::ifstream fin("input.txt");
    std::ofstream fout("output.txt");
#else
    #define fin std::cin
    #define fout std::cout
#endif
 int main() {
        int n, X;
    fin >> n >> X;
        std::vector<int> c(n);
    for (int i = 0; i < n; i++) {
        fin >> c[i];
        c[i] *= (n - i);
    }
    std::sort(c.begin(), c.end());
    int ans = 0;
    for (auto x : c) {
        if (x <= X) {
            ans += 1;
            X -= x;
        }
    }
    fout << ans << "\n";
        return 0;
}
