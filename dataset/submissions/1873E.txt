#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n, x;
    std::cin >> n >> x;
        std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
    std::sort(a.begin(), a.end());
    i64 sum = 0;
    i64 ans = 0;
    for (int i = 0; i < n; i++) {
        if (i && sum + 1LL * i * (a[i] - a[i - 1]) > x) {
            ans = a[i - 1] + (x - sum) / i;
            break;
        }
        if (i) {
            sum += 1LL * i * (a[i] - a[i - 1]);
        }
        if (i == n - 1) {
            ans = a[i] + (x - sum) / n;
        }
    }
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
