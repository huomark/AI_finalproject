#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n;
    std::cin >> n;
    std::string s;
    std::cin >> s;
        int c = std::count(s.begin(), s.end(), 'H');
    int ans = n;
    for (int i = 0; i < n; i++) {
        int cnt = std::count(s.begin(), s.begin() + c, 'T');
        ans = std::min(ans, cnt);
        std::rotate(s.begin(), s.begin()+1, s.end());
    }
    std::cout << ans << "\n";
        return 0;
}
