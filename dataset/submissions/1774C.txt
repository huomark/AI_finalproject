#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::string s;
    std::cin >> s;
        int suf = 0;
    for (int x = 2; x <= n; x++) {
        if (x > 2 && s[x - 2] != s[x - 3]) {
            suf = 1;
        } else {
            suf++;
        }
        std::cout << x - suf << " \n"[x == n];
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
