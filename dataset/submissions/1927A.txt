 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::string s;
    std::cin >> s;
        int l = 0, r = n - 1;
    while (s[l] == 'W') {
        l++;
    }
    while (s[r] == 'W') {
        r--;
    }
    int ans = r - l + 1;
    std::cout << ans << "\n";
}

}
