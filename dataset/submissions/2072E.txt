#include <bits/stdc++.h>
 using i64 = long long;
using u64 = unsigned long long;
using u32 = unsigned;
using u128 = unsigned __int128;
using i128 = __int128;
 void solve() {
    int k;
    std::cin >> k;
        std::vector<std::array<int, 2>> ans;
    int x = 0;
    int y = 0;
    while (k) {
        x++;
        int s = 1;
        while (s * (s + 1) / 2 <= k) {
            s++;
        }
        k -= s * (s - 1) / 2;
        for (int i = 1; i <= s; i++) {
            y++;
            ans.push_back({x, y});
        }
    }
    std::cout << ans.size() << "\n";
    for (auto [x, y] : ans) {
        std::cout << x << " " << y << "\n";
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
