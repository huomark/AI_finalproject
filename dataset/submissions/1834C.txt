#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::string s, t;
    std::cin >> s >> t;
        int ans = 1E9;
        for (int c = 0; c < 2; c++) {
                int diff = 0;
                for (int i = 0; i < n; i++) {
            diff += s[i] != t[i];
        }
                if (diff % 2 == c) {
            ans = std::min(ans, diff * 2);
        } else if (diff > 0) {
            ans = std::min(ans, diff * 2 - 1);
        } else {
            ans = std::min(ans, 2);
        }
                std::reverse(s.begin(), s.end());
    }
        std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
