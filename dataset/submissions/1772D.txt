#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
        int lo = 0, hi = 1E8;
        for (int i = 0; i < n - 1; i++) {
        if (a[i] < a[i + 1]) {
            hi = std::min(hi, (a[i] + a[i + 1]) / 2);
        } else if (a[i] > a[i + 1]) {
            lo = std::max(lo, (a[i] + a[i + 1] + 1) / 2);
        }
    }
        if (lo <= hi) {
        std::cout << lo << "\n";
    } else {
        std::cout << -1 << "\n";
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
