#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> l(n), r(n);
    for (int i = 0; i < n; i++) {
        std::cin >> l[i] >> r[i];
    }
        std::vector<int> o(n);
    std::iota(o.begin(), o.end(), 0);
        std::sort(o.begin(), o.end(),
        [&](int i, int j) {
            return r[i] < r[j];
        });
        int ans = 0;
    int cut = -1;
    int lastr = -1;
        for (auto i : o) {
        if (l[i] <= cut) {
            continue;
        }
        if (l[i] <= lastr) {
            ans++;
            cut = r[i];
        } else {
            lastr = r[i];
        }
    }
        std::cout << n - 2 * ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
