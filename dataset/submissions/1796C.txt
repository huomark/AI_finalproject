#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int l, r;
    std::cin >> l >> r;
        int len = std::__lg(r / l) + 1;
    int cnt = (r >> (len - 1)) - l + 1;
    if (len >= 2) {
        cnt += std::max(0, ((r / 3) >> (len - 2)) - l + 1) * (len - 1);
    }
    std::cout << len << " " << cnt << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
