#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n, m;
    std::cin >> n >> m;
        std::vector<int> a(m), b(m);
    for (int i = 0; i < m; i++) {
        std::cin >> a[i] >> b[i];
    }
        int t = 0;
    for (int i = 0; i < m; i++) {
        if (a[i] != t+1) {
            if (a[i] > t+1) {
                std::cout << t+1 << " " << 0 << "\n";
                return 0;
            }
            int cnt = std::count(a.begin(), a.end(), t) + (a[i-1] < t);
            std::cout << t << " " << cnt << "\n";
            return 0;
        }
        t = b[i];
    }
    if (t != n) {
        std::cout << t+1 << " " << 0 << "\n";
    } else {
        std::cout << "OK\n";
    }
        return 0;
}
