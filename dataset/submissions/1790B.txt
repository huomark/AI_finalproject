#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n, s, r;
    std::cin >> n >> s >> r;
        s -= n, r -= (n - 1);
        int mx = s - r;
        std::cout << mx + 1;
    for (int i = 1; i < n; i++) {
        int t = std::min(mx, r);
        r -= t;
        std::cout << " " << t + 1;
    }
    std::cout << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
