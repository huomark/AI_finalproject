#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n, w, v, u;
    std::cin >> n >> w >> v >> u;
        std::vector<int> x(n), y(n);
    for (int i = 0; i < n; i++) {
        std::cin >> x[i] >> y[i];
    }
        bool before = true;
    for (int i = 0; i < n; i++) {
        if (1LL * x[i] * u < 1LL * y[i] * v) {
            before = false;
        }
    }
        std::cout << std::fixed << std::setprecision(10);
        if (before) {
        std::cout << 1. * w / u << "\n";
        return 0;
    }
        double ans = 1. * w / u;
    for (int i = 0; i < n; i++) {
        ans = std::max(ans, 1. * x[i] / v + 1. * (w - y[i]) / u);
    }
    std::cout << ans << "\n";
        return 0;
}
