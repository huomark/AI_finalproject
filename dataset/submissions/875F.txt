#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n, m;
    std::cin >> n >> m;
        std::vector<int> f(n), c(n);
    std::iota(f.begin(), f.end(), 0);
        std::vector<std::array<int, 3>> a(m);
    for (int i = 0; i < m; i++) {
        int x, y, z;
        std::cin >> x >> y >> z;
        x--, y--;
        a[i] = {z, x, y};
    }
    std::sort(a.begin(), a.end(), std::greater());
        auto find = [&](int x) {
        while (x != f[x]) {
            x = f[x] = f[f[x]];
        }
        return x;
    };
        int ans = 0;
    for (auto [w, x, y] : a) {
        x = find(x);
        y = find(y);
                if (x == y) {
            if (c[x] == 0) {
                c[x] = 1;
                ans += w;
            }
        } else if (c[x] + c[y] < 2) {
            c[x] += c[y];
            ans += w;
            f[y] = x;
        }
    }
        std::cout << ans << "\n";
        return 0;
}
