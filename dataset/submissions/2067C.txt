#include <bits/stdc++.h>
 using i64 = long long;
using u64 = unsigned long long;
using u32 = unsigned;
using u128 = unsigned __int128;
using i128 = __int128;
 void solve() {
    int n;
    std::cin >> n;
        for (int ans = 0; ans <= 6; ans++) {
        auto s = std::to_string(n - ans);
        for (auto x : s) {
            int d = x - '0';
            if (d <= 7 && 7 - d <= ans) {
                std::cout << ans << "\n";
                return;
            }
        }
    }
    std::cout << 7 << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
