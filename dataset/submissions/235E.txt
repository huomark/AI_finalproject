#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int a, b, c;
    std::cin >> a >> b >> c;
        int N = std::max({a, b, c});
        std::vector<int> mu(N + 1);
    mu[1] = 1;
    for (int i = 1; i <= N; i++) {
        for (int j = i + i; j <= N; j += i) {
            mu[j] -= mu[i];
        }
    }
        int ans = 0;
    for (int i = 1; i <= a; i++) {
        for (int d = 1; d <= N; d++) {
            if (std::gcd(i, d) != 1) {
                continue;
            }
            int b1 = b / d;
            int c1 = c / d;
                        int s1 = 0, s2 = 0;
            for (int j = 1; j <= b1; j++) {
                if (std::gcd(i, j) == 1) {
                    s1 += b1 / j;
                }
            }
            for (int k = 1; k <= c1; k++) {
                if (std::gcd(i, k) == 1) {
                    s2 += c1 / k;
                }
            }
            ans += (a / i) * s1 * s2 * mu[d];
        }
    }
    ans &= (1 << 30) - 1;
    std::cout << ans << "\n";
        return 0;
}
