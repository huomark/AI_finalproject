#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int p;
    std::cin >> p;
        std::vector<int> dp(p + 1);
    for (int i = 1; i <= p; i++) {
        dp[i] = 1E9;
        for (int j = 2; j * (j - 1) / 2 <= i; j++) {
            dp[i] = std::min(dp[i], dp[i - j * (j - 1) / 2] + j);
        }
    }
        std::cout << dp[p] << " " << 1LL * dp[p] * (dp[p] - 1) / 2 - p << "\n";
        return 0;
}
