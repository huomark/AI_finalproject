#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> l(n);
    for (int i = 0; i < n; i++) {
        std::cin >> l[i];
    }
        for (int x = 0; x <= n; x++) {
        int cnt = 0;
        for (int i = 0; i < n; i++) {
            cnt += x < l[i];
        }
        if (cnt == x) {
            std::cout << x << "\n";
            return;
        }
    }
    std::cout << -1 << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
