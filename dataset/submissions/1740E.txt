#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n;
    std::cin >> n;
        std::vector<int> p(n);
    for (int i = 1; i < n; i++) {
        std::cin >> p[i];
        p[i]--;
    }
    p[0] = -1;
        std::vector<int> h(n, 1), dp(n);
    for (int i = n - 1; i >= 0; i--) {
        dp[i] = std::max(dp[i], h[i]);
        if (i) {
            dp[p[i]] += dp[i];
            h[p[i]] = std::max(h[p[i]], h[i] + 1);
        }
    }
        std::cout << dp[0] << "\n";
        return 0;
}
