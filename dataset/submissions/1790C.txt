#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::vector a(n, std::vector<int>(n - 1));
    std::vector<int> cnt(n);
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n - 1; j++) {
            std::cin >> a[i][j];
            a[i][j]--;
        }
        cnt[a[i][0]]++;
    }
        int x = std::max_element(cnt.begin(), cnt.end()) - cnt.begin();
        for (int i = 0; i < n; i++) {
        if (a[i][0] != x) {
            std::cout << x + 1;
            for (auto y : a[i]) {
                std::cout << " " << y + 1;
            }
            std::cout << "\n";
            return;
        }
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
