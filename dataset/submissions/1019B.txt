#include <bits/stdc++.h>
 using i64 = long long;
 int n;
 int query(int x) {
    std::cout << "? " << x % n + 1 << std::endl;
    int ans;
    std::cin >> ans;
    return ans;
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        std::cin >> n;
        int lo = 0, hi = n / 2;
    int flo = query(0) - query(n / 2);
    int fhi = -flo;
        if (flo % 2 != 0) {
        std::cout << "! " << -1 << std::endl;
        return 0;
    }
        if (flo == 0) {
        std::cout << "! " << 1 << std::endl;
        return 0;
    }
        while (true) {
        int x = (lo + hi) / 2;
        int fx = query(x) - query(x + n / 2);
        if (fx == 0) {
            std::cout << "! " << x + 1 << std::endl;
            return 0;
        }
        if ((fx > 0) == (flo > 0)) {
            lo = x;
            flo = fx;
        } else {
            hi = x;
            fhi = fx;
        }
    }
        return 0;
}
