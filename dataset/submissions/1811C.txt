#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> b(n - 1), a(n);
    for (int i = 0; i < n - 1; i++) {
        std::cin >> b[i];
    }
    for (int i = 0; i < n; i++) {
        if (i == 0) {
            a[i] = b[i];
        } else if (i == n - 1) {
            a[i] = b[i - 1];
        } else {
            a[i] = std::min(b[i - 1], b[i]);
        }
        std::cout << a[i] << " \n"[i == n - 1];
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
