#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
    i64 s = 0;
    int ans = 0;
    int max = 0;
    int first = 1;
        std::map<i64, int> cnt;
    for (int i = 0; i < n; i++) {
        if (a[i] == 0) {
            if (first) {
                first = 0;
                ans += cnt[0];
            } else {
                ans += max;
            }
            max = 0;
            cnt.clear();
        }
        s += a[i];
        max = std::max(max, ++cnt[s]);
    }
    if (first) {
        first = 0;
        ans += cnt[0];
    } else {
        ans += max;
    }
        std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
