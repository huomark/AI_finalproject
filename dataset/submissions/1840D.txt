#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
        std::sort(a.begin(), a.end());
        int lo = 0, hi = 1E9;
    while (lo < hi) {
        int x = (lo + hi) / 2;
                int i = 0;
        for (int t = 0; t < 3 && i < n; t++) {
            i = std::upper_bound(a.begin(), a.end(), 2LL * x + a[i]) - a.begin();
        }
                if (i == n) {
            hi = x;
        } else {
            lo = x + 1;
        }
    }
        std::cout << lo << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
     return 0;
}
