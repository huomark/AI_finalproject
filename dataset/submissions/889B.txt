#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n;
    std::cin >> n;
        std::vector<std::string> s(n);
    for (int i = 0; i < n; i++) {
        std::cin >> s[i];
    }
        std::vector<int> l(26, -1), r(26, -1);
    std::vector<bool> vis(26);
    for (int i = 0; i < n; i++) {
        for (int j = 1; j < s[i].size(); j++) {
            int x = s[i][j - 1] - 'a';
            int y = s[i][j] - 'a';
            if (r[x] != -1 && r[x] != y) {
                std::cout << "NO\n";
                return 0;
            }
            r[x] = y;
            if (l[y] != -1 && l[y] != x) {
                std::cout << "NO\n";
                return 0;
            }
            l[y] = x;
        }
        for (auto c : s[i]) {
            vis[c - 'a'] = true;
        }
    }
        std::string ans;
    for (int i = 0; i < 26; i++) {
        if (l[i] == -1 && vis[i]) {
            for (int j = i; j != -1; j = r[j]) {
                vis[j] = false;
                ans += 'a' + j;
            }
        }
    }
    if (std::find(vis.begin(), vis.end(), true) != vis.end()) {
        std::cout << "NO\n";
        return 0;
    }
    std::cout << ans << "\n";
        return 0;
}
