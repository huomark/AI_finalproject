#include <bits/stdc++.h>
 using i64 = long long;
using u64 = unsigned long long;
using u32 = unsigned;
using u128 = unsigned __int128;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
        std::sort(a.begin(), a.end());
        int k = -1;
    for (int i = 1; i < n; i++) {
        if (a[i] == a[i - 1]) {
            k = i;
        }
    }
    if (k == -1) {
        std::cout << -1 << "\n";
        return;
    }
        int x = a[k];
    a.erase(a.begin() + k);
    a.erase(a.begin() + k - 1);
        for (int i = 1; i < a.size(); i++) {
        if (a[i] < a[i - 1] + 2 * x) {
            std::cout << x << " " << x << " " << a[i - 1] << " " << a[i] << "\n";
            return;
        }
    }
    std::cout << -1 << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
