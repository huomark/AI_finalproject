#include <bits/stdc++.h>
 using i64 = long long;
    #ifdef ONLINE_JUDGE
    std::ifstream fin("input.txt");
    std::ofstream fout("output.txt");
#else
    #define fin std::cin
    #define fout std::cout
#endif
 int main() {
        int n;
    fin >> n;
        std::string s;
    fin >> s;
        std::vector<int> p;
    for (int i = 0; i < n; i++) {
        if (s[i] == '1') {
            p.push_back(i);
        }
    }
        for (int i = 1; i + 1 < p.size(); i++) {
        if (p[i] * 2 != p[i - 1] + p[i + 1]) {
            fout << "NO\n";
            return 0;
        }
    }
        fout << "YES\n";
        return 0;
}
