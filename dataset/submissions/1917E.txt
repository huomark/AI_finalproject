#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n, k;
    std::cin >> n >> k;
        if (k % 2 == 1) {
        std::cout << "No\n";
        return;
    }
        std::vector a(n, std::vector<int>(n));
        if (k % 4 == 0) {
        for (int i = 0; i < n; i += 2) {
            for (int j = 0; j < n; j += 2) {
                if (k > 0) {
                    a[i][j] = a[i][j + 1] = a[i + 1][j] = a[i + 1][j + 1] = 1;
                    k -= 4;
                }
            }
        }
    } else if (n == 2) {
        a[0][0] = a[1][1] = 1;
    } else if (k == 2 || k == n * n - 2) {
        std::cout << "No\n";
        return;
    } else {
        if (k > n * n / 2) {
            for (int i = 0; i < n; i++) {
                for (int j = 0; j < n; j++) {
                    a[i][j] = 1;
                }
            }
            k = n * n - k;
        }
        a[0][0] ^= 1;
        a[0][1] ^= 1;
        a[1][0] ^= 1;
        a[1][2] ^= 1;
        a[2][1] ^= 1;
        a[2][2] ^= 1;
        k -= 6;
        for (int i = 0; i < n; i += 2) {
            for (int j = 0; j < n; j += 2) {
                if (i < 4 && j < 4) {
                    continue;
                }
                if (k > 0) {
                    a[i][j] ^= 1;
                    a[i][j + 1] ^= 1;
                    a[i + 1][j] ^= 1;
                    a[i + 1][j + 1] ^= 1;
                    k -= 4;
                }
            }
        }
    }
    std::cout << "Yes\n";
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            std::cout << a[i][j] << " \n"[j == n - 1];
        }
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
