#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        i64 sum = 0;
    int odd = 0;
    for (int i = 1; i <= n; i++) {
        int a;
        std::cin >> a;
                sum += a;
        odd += a % 2;
                int res;
        if (odd == 1 && i == 1) {
            res = 0;
        } else {
            res = odd / 3;
            if (odd % 3 == 1) {
                res += 1;
            }
        }
        std::cout << sum - res << " \n"[i == n];
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
