#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n, m, k, H;
    std::cin >> n >> m >> k >> H;
        int ans = 0;
    for (int i = 0; i < n; i++) {
        int h;
        std::cin >> h;
                if (h != H && (h - H) % k == 0 && std::abs(h - H) < m * k) {
            ans++;
        }
    }
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
