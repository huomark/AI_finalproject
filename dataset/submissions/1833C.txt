 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
    std::sort(a.begin(), a.end());
        for (int c = 0; c < 2; c++) {
        bool odd = false;
        bool ok = true;
        for (int i = 0; i < n; i++) {
            if (i && a[i] > a[i - 1] && a[i - 1] % 2) {
                odd = true;
            }
            if (a[i] % 2 != c && !odd) {
                ok = false;
            }
        }
        if (ok) {
            std::cout << "YES\n";
            return;
        }
    }
    std::cout << "NO\n";
}
 int main() {
