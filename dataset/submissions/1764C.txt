#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::map<int, int> cnt;
        std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
        cnt[a[i]]++;
    }
        i64 ans = 1;
        int s = 0;
        for (auto [_, x] : cnt) {
        s += x;
        ans = std::max(ans, 1LL * s * (n - s));
    }
        if (cnt.size() == 1) {
        ans = n / 2;
    }
        std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
