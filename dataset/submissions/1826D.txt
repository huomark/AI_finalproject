#include <bits/stdc++.h>
 using i64 = long long;
 constexpr int inf = 1E9;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
        std::array<int, 4> dp{-inf};
    dp[0] = 0;
    for (int i = 0; i < n; i++) {
        dp[1] -= 1;
        dp[2] -= 1;
        for (int j = 3; j; j--) {
            dp[j] = std::max(dp[j], dp[j - 1] + a[i]);
        }
    }
    std::cout << dp[3] << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
