#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n;
    std::cin >> n;
        std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
    std::sort(a.begin(), a.end());
        const int m = a[n - 1];
        std::vector<int> mu(m + 1);
    mu[1] = 1;
    for (int i = 1; i <= m; i++) {
        for (int j = 2 * i; j <= m; j += i) {
            mu[j] -= mu[i];
        }
    }
        std::vector<std::vector<int>> divs(m + 1);
    for (int i = 1; i <= m; i++) {
        for (int j = i; j <= m; j += i) {
            divs[j].push_back(i);
        }
    }
        i64 ans = 0, cur = 0;
        std::vector<int> cl(m + 1), cr(m + 1);
    for (int i = 0; i < n; i++) {
        for (auto d : divs[a[i]]) {
            cr[d]++;
        }
    }
        for (int i = 0; i < n; i++) {
        for (auto d : divs[a[i]]) {
            cur -= mu[d] * cl[d];
            cr[d]--;
        }
        ans += cur;
        for (auto d : divs[a[i]]) {
            cur += mu[d] * cr[d];
            cl[d]++;
        }
    }
        std::cout << ans << "\n";
        return 0;
}
