#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n;
    i64 j;
    std::cin >> n >> j;
        i64 mx, my;
    std::cin >> mx >> my;
        std::vector<i64> x(n), y(n);
    for (int i = 0; i < n; i++) {
        std::cin >> x[i] >> y[i];
    }
        j %= 2 * n;
    for (int i = 0; i < j; i++) {
        mx = 2 * x[i % n] - mx;
        my = 2 * y[i % n] - my;
    }
    std::cout << mx << " " << my << "\n";
        return 0;
}
