#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int x, y, k;
    std::cin >> x >> y >> k;
        int ans;
    if (x > y) {
        ans = x;
    } else {
        ans = y + std::max(0, y - x - k);
    }
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
