#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    std::string s;
    std::cin >> s;
        int n = s.size();
        int ans = 0;
    for (int d = 1; d <= n / 2; d++) {
        int v = 0;
        for (int i = n - d - 1; i >= 0; i--) {
            v = (s[i] == s[i + d] || s[i] == '?' || s[i + d] == '?') ? v + 1 : 0;
            if (v == d) {
                ans = d;
                break;
            }
        }
    }
    std::cout << 2 * ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
