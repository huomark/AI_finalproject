#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int A, B, C;
    std::cin >> A >> B >> C;
        std::cout << std::fixed << std::setprecision(10);
        if (A != 0) {
        if (A < 0) {
            A = -A;
            B = -B;
            C = -C;
        }
        i64 t = 1LL * B * B - 4LL * A * C;
        if (t < 0) {
            std::cout << 0 << "\n";
            return 0;
        }
        if (t == 0) {
            std::cout << 1 << "\n";
            std::cout << -1. * B / (2 * A) << "\n";
        } else {
            std::cout << 2 << "\n";
            std::cout << (-B - std::sqrt(t)) / (2 * A) << "\n";
            std::cout << (-B + std::sqrt(t)) / (2 * A) << "\n";
        }
    } else if (B != 0) {
        std::cout << 1 << "\n";
        std::cout << 1. * -C / B << "\n";
    } else {
        if (C == 0) {
            std::cout << -1 << "\n";
        } else {
            std::cout << 0 << "\n";
        }
    }
        return 0;
}
