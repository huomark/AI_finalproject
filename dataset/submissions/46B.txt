#include <bits/stdc++.h>
 using i64 = long long;
 std::string s[] = {"S", "M", "L", "XL", "XXL"};
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        std::map<std::string, int> id;
    for (int i = 0; i < 5; i++) {
        id[s[i]] = i;
    }
        int cnt[5] {};
    for (int i = 0; i < 5; i++) {
        std::cin >> cnt[i];
    }
        int k;
    std::cin >> k;
        while (k--) {
        std::string t;
        std::cin >> t;
        int x = id[t];
                int j = -1;
        for (int i = 0; i < 5; i++) {
            if (cnt[i] && (j == -1 || std::abs(i-x) <= std::abs(j-x))) {
                j = i;
            }
        }
        cnt[j]--;
        std::cout << s[j] << "\n";
    }
        return 0;
}
