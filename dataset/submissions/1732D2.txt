#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int q;
    std::cin >> q;
        std::map<i64, i64> f;
    std::map<i64, std::set<i64>> g, d;
    std::set<i64> s{0};
        for (int i = 0; i < q; i++) {
        char o;
        i64 x;
        std::cin >> o >> x;
        if (o == '+') {
            s.insert(x);
            for (auto y : d[x]) {
                g[y].erase(x);
            }
        } else if (o == '-') {
            s.erase(x);
            for (auto y : d[x]) {
                g[y].insert(x);
            }
        } else {
            if (g[x].empty()) {
                while (s.contains(f[x])) {
                    f[x] += x;
                    d[f[x]].insert(x);
                }
                std::cout << f[x] << "\n";
            } else {
                std::cout << *g[x].begin() << "\n";
            }
        }
    }
        return 0;
}
