#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n;
    std::cin >> n;
        std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
        int ans = 0;
    for (int t = 0; t < 20; t++) {
        std::vector<int> s(n + 1);
        for (int i = 0; i < n; i++) {
            s[i + 1] = s[i] ^ a[i];
        }
        std::vector<int> r(n + 1);
        r[n] = n;
        for (int i = n - 1; i >= 0; i--) {
            r[i] = a[i] & 1 ? r[i + 1] : i;
        }
        std::vector<int> lst(1 << (20 - t), -1);
        std::vector<int> f(n + 1);
        for (int i = n; i >= 0; i--) {
            f[i] = i;
            if (lst[s[i]] != -1) {
                int j = lst[s[i]];
                if (r[i] >= j) {
                    f[i] = f[j];
                }
            }
            ans = std::max(ans, f[i] - i);
            lst[s[i]] = i;
        }
        for (auto &x : a) {
            x /= 2;
        }
    }
        std::cout << ans << "\n";
        return 0;
}
