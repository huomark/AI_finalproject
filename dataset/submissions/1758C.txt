#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n, x;
    std::cin >> n >> x;
        if (n % x != 0) {
        std::cout << -1 << "\n";
        return;
    }
        std::vector<int> p(n);
    std::iota(p.begin(), p.end(), 1);
    p[0] = x;
    p[n - 1] = 1;
        for (int i = 2 * x; i <= n; i += x) {
        if (n % i == 0) {
            p[x - 1] = i;
            x = i;
        }
    }
        for (int i = 0; i < n; i++) {
        std::cout << p[i] << " \n"[i == n - 1];
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
