#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n;
    std::cin >> n;
        int x = 0, y = 1;
    std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
    for (int i = 0; i < n; i++) {
        int j = (i + 1) % n;
        if (std::abs(a[i] - a[j]) < std::abs(a[x] - a[y])) {
            x = i;
            y = j;
        }
    }
    std::cout << x + 1 << " " << y + 1 << "\n";
        return 0;
}
