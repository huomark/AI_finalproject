#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::string s;
    std::cin >> s;
        std::array<int, 26> cnt{};
    for (auto c : s) {
        cnt[c - 'a'] += 1;
    }
        int ans = n - 2 * std::min(n / 2, n - *std::max_element(cnt.begin(), cnt.end()));
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
