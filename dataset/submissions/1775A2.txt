#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    std::string s;
    std::cin >> s;
        int n = s.size();
        int pos = 0;
    for (int i = 1; i < n - 1; i++) {
        if (s[i] == 'a') {
            pos = i;
            break;
        }
    }
        if (pos) {
        std::cout << s.substr(0, pos) << " " << 'a' << " " << s.substr(pos + 1) << "\n";
    } else {
        std::cout << s[0] << " " << s.substr(1, n - 2) << " " << s[n - 1] << "\n";
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
