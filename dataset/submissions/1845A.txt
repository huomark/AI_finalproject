 void solve() {

            std::cout << 1 << " \n"[i == n - 1];

    }
        if (k > 1 && n % 2 == 0) {
        std::cout << "YES\n";
        std::cout << n / 2 << "\n";

            std::cout << 2 << " \n"[i == n / 2 - 1];
        }

    }

            std::cout << 2 << " ";
        }

}
