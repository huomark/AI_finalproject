#include <bits/stdc++.h>
 using i64 = long long;
 i64 F[50];
 void solve() {
    int n;
    i64 x, y;
    std::cin >> n >> x >> y;
        while (n > 0) {
        if (y <= F[n - 1]) {
                    } else if (y > F[n]) {
            y -= F[n];
        } else {
            std::cout << "NO\n";
            return;
        }
        std::swap(x, y);
        n -= 1;
    }
    std::cout << "YES\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        F[0] = F[1] = 1;
    for (int i = 2; i < 50; i++) {
        F[i] = F[i - 2] + F[i - 1];
    }
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
