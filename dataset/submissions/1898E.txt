#include <bits/stdc++.h>
 using i64 = long long;
 constexpr i64 inf = 1E18;
 void solve() {
    int n, m;
    std::cin >> n >> m;
        std::string s, t;
    std::cin >> s >> t;
        std::array<std::vector<int>, 26> vec;
    for (int i = n - 1; i >= 0; i--) {
        vec[s[i] - 'a'].push_back(i);
    }
        for (int i = 0; i < m; i++) {
        int x = t[i] - 'a';
        if (vec[x].empty()) {
            std::cout << "NO\n";
            return;
        }
        int p = vec[x].back();
        vec[x].pop_back();
        for (int j = 0; j < x; j++) {
            while (!vec[j].empty() && vec[j].back() < p) {
                vec[j].pop_back();
            }
        }
    }
    std::cout << "YES\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
