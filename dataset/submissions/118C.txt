#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n, k;
    std::cin >> n >> k;
        std::string s;
    std::cin >> s;
        std::string t;
    int ans = 1E9;
        std::vector<int> f[10];
    for (int i = 0; i < n; i++) {
        f[s[i] - '0'].push_back(i);
    }
    for (int x = 0; x <= 9; x++) {
        std::string a = s;
        int cnt = 0;
        for (int d = 0; d <= 9; d++) {
            if (x + d <= 9) {
                for (auto i : f[x + d]) {
                    if (cnt < k) {
                        cnt++;
                        a[i] = '0' + x;
                    }
                }
            }
            if (x >= d && d) {
                std::reverse(f[x - d].begin(), f[x - d].end());
                for (auto i : f[x - d]) {
                    if (cnt < k) {
                        cnt++;
                        a[i] = '0' + x;
                    }
                }
                std::reverse(f[x - d].begin(), f[x - d].end());
            }
        }
        int res = 0;
        for (int i = 0; i < n; i++) {
            res += std::abs(s[i] - a[i]);
        }
        if (res < ans || (res == ans && a < t)) {
            ans = res;
            t = a;
        }
    }
        std::cout << ans << "\n";
    std::cout << t << "\n";
        return 0;
}
