#include <bits/stdc++.h>
 using i64 = long long;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        std::string s;
    std::cin >> s;
        int n = s.size();
        std::vector<int> f(n + 1);
    for (int i = 0; i < n; i++) {
        f[i + 1] = f[i] + (s[i] == '(' ? 1 : -1);
    }
        int min = *std::min_element(f.begin(), f.end());
        int ans = n - (f[0] - min) - (f[n] - min);
    std::cout << ans << "\n";
        return 0;
}
