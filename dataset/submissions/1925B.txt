#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n, x;
    std::cin >> n >> x;
        int ans = n;
    for (int i = 1; i * i <= n; i++) {
        if (n % i == 0) {
            if (i >= x) {
                ans = std::min(ans, i);
            }
            if (n / i >= x) {
                ans = std::min(ans, n / i);
            }
        }
    }
    std::cout << n / ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
