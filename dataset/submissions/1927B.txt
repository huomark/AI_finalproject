#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::string s;
        std::array<int, 26> cnt{};
    for (int i = 0; i < n; i++) {
        int a;
        std::cin >> a;
                for (int x = 0; x < 26; x++) {
            if (cnt[x] == a) {
                s += 'a' + x;
                cnt[x]++;
                break;
            }
        }
    }
        std::cout << s << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
