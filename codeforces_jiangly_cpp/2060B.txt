#include <bits/stdc++.h>
 using i64 = long long;
using u64 = unsigned long long;
using u32 = unsigned;
using u128 = unsigned __int128;
 void solve() {
    int n, m;
    std::cin >> n >> m;
        bool ok = true;
    std::vector<int> p(n);
    for (int i = 0; i < n; i++) {
        int x = 0;
        for (int j = 0; j < m; j++) {
            int a;
            std::cin >> a;
            if (j == 0) {
                x = a % n;
            }
            if (a % n != x) {
                ok = false;
            }
        }
        p[x] = i;
    }
    if (ok) {
        for (int i = 0; i < n; i++) {
            std::cout << p[i] + 1 << " \n"[i == n - 1];
        }
    } else {
        std::cout << -1 << "\n";
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
