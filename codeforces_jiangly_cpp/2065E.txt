#include <bits/stdc++.h>
 using i64 = long long;
using u64 = unsigned long long;
using u32 = unsigned;
using u128 = unsigned __int128;
 void solve() {
    int n, m, k;
    std::cin >> n >> m >> k;
        if (std::abs(n - m) > k || std::max(n, m) < k) {
        std::cout << -1 << "\n";
        return;
    }
        int t = 0;
    if (n < m) {
        std::swap(n, m);
        t = 1;
    }
    std::string s;
    while (n > 0 || m > 0) {
        for (int i = 0; i < k && n; i++) {
            n--;
            s += '0' ^ t;
        }
        for (int i = 0; i < k && m; i++) {
            m--;
            s += '1' ^ t;
        }
    }
    std::cout << s << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
