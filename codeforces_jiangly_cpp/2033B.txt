#include <bits/stdc++.h>
 using i64 = long long;
using u64 = unsigned long long;
using u32 = unsigned;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> f(n + n - 1);
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            int a;
            std::cin >> a;
            int k = i - j + n - 1;
            f[k] = std::max(f[k], -a);
        }
    }
        int ans = std::accumulate(f.begin(), f.end(), 0);
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
