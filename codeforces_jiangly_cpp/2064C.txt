#include <bits/stdc++.h>
 using i64 = long long;
using u64 = unsigned long long;
using u32 = unsigned;
using u128 = unsigned __int128;
using i128 = __int128;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
        i64 suf = 0;
    i64 pre = 0;
    for (int i = 0; i < n; i++) {
        if (a[i] < 0) {
            suf += a[i];
        }
    }
    i64 ans = -suf;
    for (int i = 0; i < n; i++) {
        if (a[i] < 0) {
            suf -= a[i];
        } else {
            pre += a[i];
        }
        ans = std::max(ans, pre - suf);
    }
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
