#include <bits/stdc++.h>
 using i64 = long long;
using u64 = unsigned long long;
using u32 = unsigned;
using u128 = unsigned __int128;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n, k, q;
    std::cin >> n >> k >> q;
        std::vector a(k, std::vector<int>(n));
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < k; j++) {
            std::cin >> a[j][i];
        }
    }
        for (int j = 0; j < k; j++) {
        for (int i = 1; i < n; i++) {
            a[j][i] |= a[j][i - 1];
        }
    }
        for (int i = 0; i < q; i++) {
        int m;
        std::cin >> m;
                int L = 0, R = n - 1;
        for (int j = 0; j < m; j++) {
            int r, c;
            char o;
            std::cin >> r >> o >> c;
            r--;
                        if (o == '>') {
                int x = std::upper_bound(a[r].begin(), a[r].end(), c) - a[r].begin();
                L = std::max(L, x);
            } else {
                int x = std::lower_bound(a[r].begin(), a[r].end(), c) - a[r].begin();
                R = std::min(R, x - 1);
            }
        }
                if (L <= R) {
            std::cout << L + 1 << "\n";
        } else {
            std::cout << -1 << "\n";
        }
    }
        return 0;
}
