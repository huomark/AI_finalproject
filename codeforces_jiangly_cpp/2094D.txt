#include <bits/stdc++.h>
 using i64 = long long;
using u64 = unsigned long long;
using u32 = unsigned;
using u128 = unsigned __int128;
 void solve() {
    std::string p, s;
    std::cin >> p >> s;
        int i = 0;
    for (int l = 0, r = 0; l < p.size(); l = r) {
        while (r < p.size() && p[l] == p[r]) {
            r++;
        }
        if (i == s.size() || s[i] != p[l]) {
            std::cout << "NO\n";
            return;
        }
        int j = i;
        while (j < s.size() && s[i] == s[j]) {
            j++;
        }
        if (j - i < r - l || j - i > 2 * (r - l)) {
            std::cout << "NO\n";
            return;
        }
        i = j;
    }
    if (i == s.size()) {
        std::cout << "YES\n";
    } else {
        std::cout << "NO\n";
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
