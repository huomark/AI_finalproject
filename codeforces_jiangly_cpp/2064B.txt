#include <bits/stdc++.h>
 using i64 = long long;
using u64 = unsigned long long;
using u32 = unsigned;
using u128 = unsigned __int128;
using i128 = __int128;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> a(n);
    std::vector<int> cnt(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
        a[i]--;
        cnt[a[i]]++;
    }
        int l = 0, r = 0;
    int len = 0;
    for (int i = 0; i < n; i++) {
        if (cnt[a[i]] > 1) {
            len = 0;
        } else {
            len++;
        }
        if (len > r - l) {
            r = i + 1;
            l = i + 1 - len;
        }
    }
        if (l == r) {
        std::cout << 0 << "\n";
    } else {
        std::cout << l + 1 << " " << r << "\n";
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
