#include <bits/stdc++.h>
 using i64 = long long;
using u64 = unsigned long long;
using u32 = unsigned;
using u128 = unsigned __int128;
using i128 = __int128;
 void solve() {
    int n, x;
    std::cin >> n >> x;
        int a = 0;
    int o = 0;
    while (a < n && ((a & x) == a)) {
        o |= a;
        a++;
    }
    if (a == n && o != x) {
        a--;
    }
    for (int i = 0; i < n; i++) {
        std::cout << (i < a ? i : x) << " \n"[i == n - 1];
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
