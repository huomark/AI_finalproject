 bool isprime(int n) {
    if (n < 2) {
        return false;
    }
    for (int i = 2; i * i <= n; i++) {
        if (n % i == 0) {
            return false;
        }
    }
    return true;
}
 void solve() {
    int x, y;
    std::cin >> x >> y;
        if (x > 1 && y > 1) {
        std::cout << "NO\n";

    }
        int a = x;
    for (int i = 1; i < y; i++) {
        a = a * 10 + x;
    }
        if (isprime(a)) {
        std::cout << "YES\n";

    }
}
