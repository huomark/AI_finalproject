#include <bits/stdc++.h>
 using i64 = long long;
using u64 = unsigned long long;
using u32 = unsigned;
using u128 = unsigned __int128;
 void solve() {
    int n, k;
    std::cin >> n >> k;
        std::vector<int> a(n), cnt(k);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
        a[i] %= k;
        cnt[a[i]]++;
    }
    for (int i = 0; i < n; i++) {
        if (cnt[a[i]] == 1) {
            std::cout << "YES\n";
            std::cout << i + 1 << "\n";
            return;
        }
    }
    std::cout << "NO\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
