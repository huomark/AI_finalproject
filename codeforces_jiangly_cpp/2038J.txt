#include <bits/stdc++.h>
 using i64 = long long;
using u64 = unsigned long long;
using u32 = unsigned;
using u128 = unsigned __int128;
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int n;
    std::cin >> n;
        std::vector<char> t(n);
    std::vector<int> x(n);
    for (int i = 0; i < n; i++) {
        std::cin >> t[i] >> x[i];
    }
        for (int i = 0; i < n; i++) {
        if (t[i] == 'B') {
            int s = 0;
            for (int j = 0; j < i; j++) {
                if (t[j] == 'P') {
                    s += x[j];
                } else {
                    s = std::max(0, s - x[j]);
                }
            }
            if (s < x[i]) {
                std::cout << "YES\n";
            } else {
                std::cout << "NO\n";
            }
        }
    }
        return 0;
}
