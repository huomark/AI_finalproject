#include <bits/stdc++.h>
 using i64 = long long;
using u64 = unsigned long long;
using u32 = unsigned;
using u128 = unsigned __int128;
 void solve() {
    std::string s;
    std::cin >> s;
        int q;
    std::cin >> q;
        int n = s.size();
        int ans = 0;
    for (int i = 0; i < n - 3; i++) {
        ans += s.substr(i, 4) == "1100";
    }
        for (int i = 0; i < q; i++) {
        int p;
        char v;
        std::cin >> p >> v;
        p--;
                for (int j = std::max(0, p - 3); j <= std::min(p, n - 4); j++) {
            ans -= s.substr(j, 4) == "1100";
        }
        s[p] = v;
        for (int j = std::max(0, p - 3); j <= std::min(p, n - 4); j++) {
            ans += s.substr(j, 4) == "1100";
        }
        std::cout << (ans ? "YES" : "NO") << "\n";
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
