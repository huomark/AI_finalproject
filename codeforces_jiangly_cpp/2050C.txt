#include <bits/stdc++.h>
 using i64 = long long;
using u64 = unsigned long long;
using u32 = unsigned;
using u128 = unsigned __int128;
 void solve() {
    std::string n;
    std::cin >> n;
        int f = 1;
    for (auto c : n) {
        int d = c - '0';
        int x = d * d;
        int nf = f << d;
        if (x < 10) {
            nf |= f << x;
        }
        nf |= nf >> 9;
        nf &= (1 << 9) - 1;
        f = nf;
    }
        if (f & 1) {
        std::cout << "YES\n";
    } else {
        std::cout << "NO\n";
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
