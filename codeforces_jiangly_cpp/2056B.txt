#include <bits/stdc++.h>
 using i64 = long long;
using u64 = unsigned long long;
using u32 = unsigned;
using u128 = unsigned __int128;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<std::string> g(n);
    for (int i = 0; i < n; i++) {
        std::cin >> g[i];
    }
        std::vector<int> p(n, -1);
    for (int i = n - 1; i >= 0; i--) {
        int k = 0;
        for (int j = 0; j < i; j++) {
            k += (g[j][i] == '1');
        }
        for (int j = 0; j < n; j++) {
            if (p[j] == -1) {
                if (k == 0) {
                    p[j] = i;
                    break;
                }
                k--;
            }
        }
    }
    for (int i = 0; i < n; i++) {
        std::cout << p[i] + 1 << " \n"[i == n - 1];
    }
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
