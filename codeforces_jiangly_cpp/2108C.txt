 using i64 = long long;

 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> a(n);
    for (int i = 0; i < n; i++) {
        std::cin >> a[i];
    }
        a.erase(std::unique(a.begin(), a.end()), a.end());
    n = a.size();
        int ans = 0;
    for (int i = 0; i < n; i++) {
        if ((i == 0 || a[i] > a[i - 1]) && (i + 1 == n || a[i] > a[i + 1])) {
            ans++;
        }
    }
    std::cout << ans << "\n";
}
