#include <bits/stdc++.h>
 using i64 = long long;
using u64 = unsigned long long;
using u32 = unsigned;
using u128 = unsigned __int128;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> c(n), f(n);
    for (int i = 0; i < n; i++) {
        std::cin >> c[i];
        c[i]--;
        f[c[i]]++;
    }
    int ans = std::count(f.begin(), f.end(), 1);
    ans = (ans + 1) / 2 * 2;
    for (int i = 0; i < n; i++) {
        ans += (f[i] >= 2);
    }
    std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
