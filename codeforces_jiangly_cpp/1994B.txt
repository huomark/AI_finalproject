 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::string s, t;
    std::cin >> s >> t;
        bool ok = true;
    for (int i = 0; i < n; i++) {
        if (s[i] == '1') {
            break;
        }
        if (t[i] == '1') {
            ok = false;

        }
    }

}
