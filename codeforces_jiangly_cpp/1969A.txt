#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    int n;
    std::cin >> n;
        std::vector<int> p(n);
    for (int i = 0; i < n; i++) {
        std::cin >> p[i];
        p[i]--;
    }
        int ans = 3;
    for (int i = 0; i < n; i++) {
        if (p[p[i]] == i) {
            ans = 2;
        }
    }
    std::cout << ans << "\n";
}

    std::cin >> t;

        solve();
    }

}
