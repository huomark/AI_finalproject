#include <bits/stdc++.h>
 using i64 = long long;
 void solve() {
    i64 n, k;
    std::cin >> n >> k;
        if (n != k && k > (n + 1) / 2) {
        std::cout << "NO\n";
        return;
    }
    std::cout << "YES\n";
    std::cout << 2 << "\n";
    std::cout << n - k + 1 << " " << 1 << "\n";
}
 int main() {

    std::cin.tie(nullptr);

    std::cin >> t;

    }

}
