#include <bits/stdc++.h>
 using i64 = long long;
using u64 = unsigned long long;
using u32 = unsigned;
using u128 = unsigned __int128;
 void solve() {
    int x;
    i64 m;
    std::cin >> x >> m;
    m++;
        int n = 1;
    while (n <= x) {
        n *= 2;
    }
        i64 ans = 0;
    for (i64 y = 1; y < std::min<i64>(n, m); y++) {
        i64 z = x ^ y;
        if (z != 0 && (x % z == 0 || y % z == 0)) {
            ans++;
        }
    }
        std::cout << ans << "\n";
}
 int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
        int t;
    std::cin >> t;
        while (t--) {
        solve();
    }
        return 0;
}
